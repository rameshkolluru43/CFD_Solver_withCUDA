19void Condition_For_MOVERS(double & d_U, double & d_F,double & L_Max, double & L_Min,double & Alpha)
20{
21 double epsilon = 1e-10;
22//***************************************************************************************************************
23
24 if(fabs(d_F) < epsilon and fabs(d_U) > epsilon) // Condition for discontinuity.... across which Flux is zero
25 {
26 Alpha = 0.0;
27 }
28 else if(fabs(d_F) < epsilon and fabs(d_U) < epsilon)
29 {
30 Alpha = L_Min;
31 }
32 else if (fabs(d_F) > epsilon and fabs(d_U) > epsilon)
33 {
34 Alpha = fabs(d_F/d_U);
35// Wave speed correction limiting the L_min<=|alpha|<=L_Max
36 if(Alpha >= L_Max)
37 Alpha = Sign(Alpha)*L_Max;
38 else if (Alpha <= L_Min)
39 Alpha = Sign(Alpha)*L_Min;
40 }
41 else
42 Alpha = L_Min;
