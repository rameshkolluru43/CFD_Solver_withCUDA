306void Viscous_Flux_on_Face(const int &Cell_No, const int &Face_No)
307{
308 // cout<<"Calculating Viscous flux for Cell No\t"<<Cell_No<<"\t on face \t"<<Face_No<<endl;
309 double u11, u12, u21, u22, T11, T12, T21, T22;

313 Qx = 0.0;

319 ny = Cell_Face_Normals[Cell_No][index + 1];
320 dl = Cell_Face_Areas[Cell_No][Face_No];

322 // Fetching the Corresponding Neighbours for the given cell and given face

371 Cells_Viscous_Flux[Cell_No][3] += ((T11 * v1 + T12 * v2 + Qx) * nx * dl + (T21 * v1 + T22 * v2 + Qy) * ny * dl);
