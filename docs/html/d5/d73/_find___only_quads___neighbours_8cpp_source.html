17// Structure to store a sorted edge representation
18struct Edge {
19 int p1, p2;
20 Edge(int a, int b) {
21 if (a < b) { p1 = a; p2 = b; }
22 else { p1 = b; p2 = a; }
23 }
24 bool operator<(const Edge& other) const {
25 return (p1 < other.p1) || (p1 == other.p1 && p2 < other.p2);
26 }
27};
