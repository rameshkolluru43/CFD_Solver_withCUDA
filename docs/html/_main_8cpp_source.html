94// Function to read parameters from a configuration file
95map<string, string> ReadConfigFile(const string& filename) {
96 map<string, string> config;
97 ifstream file(filename);
98
99 if (!file.is_open()) {
100 cerr << "Error: Unable to open configuration file " << filename << endl;
101 exit(EXIT_FAILURE);
102 }
103
104 string line;
105 while (getline(file, line)) {
106 if (line.empty() || line[0] == '#') continue; // Skip empty lines and comments
107 size_t delimiter = line.find('=');
108 if (delimiter == string::npos) continue;
109
110 string key = line.substr(0, delimiter);
111 string value = line.substr(delimiter + 1);
112
113 // Trim whitespace from key and value
114 key.erase(key.find_last_not_of(" \t") + 1);
115 value.erase(0, value.find_first_not_of(" \t"));
116 value.erase(value.find_last_not_of(" \t") + 1);
117
118 config[key] = value;
119 }
120 file.close();
121 return config;
122}
