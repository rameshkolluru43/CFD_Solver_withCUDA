131map<string, string> ReadConfigFile(const string &filename)
132{
133 map<string, string> config;
134 ifstream file(filename);
135
136 if (!file.is_open())
137 {
138 cerr << "Error: Unable to open configuration file " << filename << endl;
139 exit(EXIT_FAILURE);
140 }
141
142 string line;
143 while (getline(file, line))
144 {
145 if (line.empty() || line[0] == '#')
146 continue; // Skip empty lines and comments
147 size_t delimiter = line.find('=');
148 if (delimiter == string::npos)
149 continue;
150
151 string key = line.substr(0, delimiter);
152 string value = line.substr(delimiter + 1);
153
154 // Trim whitespace from key and value
155 key.erase(key.find_last_not_of(" \t") + 1);
156 value.erase(0, value.find_first_not_of(" \t"));
157 value.erase(value.find_last_not_of(" \t") + 1);
158
159 config[key] = value;
160 }
161 file.close();
162 return config;
163}
