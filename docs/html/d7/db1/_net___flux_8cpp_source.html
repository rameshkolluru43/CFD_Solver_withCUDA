92 // This function iterates over all physical cells and calculates the net flux for each cell
93 // based on the specified dissipation type. It initializes the net flux for each cell to zero,
94 // applies the appropriate flux calculation method for all faces of the cell, and evaluates
95 // the time step for the current cell.
96 // The dissipation type determines which flux calculation method is used:
97 // 1: LLF (Local Lax-Friedrichs)
98 // 2: MOVERS
99 // 3: ROE (Roe's method)
100 // 4: RICCA (Ricca's method)
101 // 5: MOVERS_NWSC (MOVERS with no wall shear correction)
102 // The function also evaluates the time step for each cell after computing the flux.
103 // The function supports multiple dissipation types, each corresponding to a specific
104 // flux calculation method.
105 for (int Current_Cell_No = 0; Current_Cell_No < No_Physical_Cells; Current_Cell_No++)
106 {
107
108 for (int i = 0; i < 4; i++)
109 {
110 Cells_Net_Flux[Current_Cell_No][i] = 0.0;
111 }
112
113 switch (Dissipation_Type)
114 {
115 case 1:
116 Calculate_Flux_For_All_Faces(Current_Cell_No, LLF);
117 break;
118 case 2:
119 Calculate_Flux_For_All_Faces(Current_Cell_No, MOVERS);
120 break;
121 case 3:
122 Calculate_Flux_For_All_Faces(Current_Cell_No, ROE);
123 break;
124 case 4:
125 Calculate_Flux_For_All_Faces(Current_Cell_No, RICCA);
126 break;
127 case 5:
128 Calculate_Flux_For_All_Faces(Current_Cell_No, MOVERS_NWSC);
129 break;
130 }
131
132 Evaluate_Time_Step(Current_Cell_No);
133 }
