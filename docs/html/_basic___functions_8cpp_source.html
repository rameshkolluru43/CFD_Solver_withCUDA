231//This Function calculates U from the primitive variables. This takes three arguments first argument can be either pressure or density, velocity Vector , temperature and an integer which is flag to identify whether the first argument is a pressure or density.
232void Calculate_Computational_Variables(const double & var,const V_D & V,const double & var1, const int & i)
233{
234 double temp_rho=0.0,vmag=0.0,v1=0.0,v2=0.0,T=0.0,P=0.0;
235// cout<<i<<"\t"<<var<<"\t"<<var1<<"\t"<<V[0]<<"\t"<<V[1]<<endl;
236// Zero Initialization before starting evaluation.
237 Global_U[0] = 0.0; Global_U[1] = 0.0; Global_U[2] = 0.0; Global_U[3] = 0.0;
238
239//This switch statement is used because var can be pressure or density, default value assumes density is passed other wise if Pressure is passed then i should take the value of 1.
240 switch(i)
241 {
242 case 1:
243 T = var1;
244 temp_rho = var/(R_GC*T); // If Pressure and Temperature are given
245 break;
246 case 2: // if Pressure and Density are given
247 P = var;
248 temp_rho = var1;
249 break;
250 default: // if density and Temperature are given
251 temp_rho = var;
252 T = var1;
253 P = temp_rho*R_GC*T;
254 break;
255 }
256 v1 = V[0]; v2 =V[1];
257 vmag = 0.5*(v1*v1 + v2*v2);
258 Global_U[0] = temp_rho;
259 Global_U[1] = temp_rho*v1;
260 Global_U[2] = temp_rho*v2;
261 Global_U[3] =((P/gamma_M_1) + temp_rho*vmag);
262// Print(Global_U);
263}
