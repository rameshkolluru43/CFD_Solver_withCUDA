283void Viscous_Flux_on_Face(const int & Cell_No,const int &Face_No)
284{
285 // cout<<"Calculating Viscous flux for Cell No\t"<<Cell_No<<"\t on face \t"<<Face_No<<endl;

311 mu = 0.5*(Primitive_Cells[Cell_No][8] + Primitive_Cells[N_Cell_No][8]);

333 Qy = K1*T_Gradient[1];

338 Cells_Viscous_Flux[Cell_No][3] += ((T11*v1 + T12*v2 + Qx)*nx*dl + (T21*v1 + T22*v2 + Qy)*ny*dl);
