6// Kernel for calculating cell volumes using Gauss theorem (divergence theorem)
7__global__ void calculate_cell_volumes_kernel(
8 const double* cell_coords, // [num_cells * max_points * 3] - Cell vertex coordinates
9 const double* face_centers, // [num_faces * 3] - Face center coordinates
10 const double* face_areas, // [num_faces] - Face areas
11 const double* face_normals, // [num_faces * 3] - Face normal vectors
12 const int* cell_faces, // [num_cells * max_faces] - Face indices for each cell
13 const int* num_faces_per_cell,// [num_cells] - Number of faces per cell
14 double* cell_volumes, // [num_cells] - Output cell volumes
15 int num_cells,
16 int max_faces
17) {
18 int idx = blockIdx.x * blockDim.x + threadIdx.x;
19 if (idx >= num_cells) return;
20
21 int num_faces = num_faces_per_cell[idx];
22 double volume = 0.0;
23
24 // Use Gauss theorem: V = (1/3) * sum(r_face · n_face * A_face)
25 for (int f = 0; f < num_faces; f++) {
26 int face_id = cell_faces[idx * max_faces + f];
27 if (face_id >= 0) {
28 // Face center coordinates
29 double xf = face_centers[face_id * 3 + 0];
30 double yf = face_centers[face_id * 3 + 1];
31 double zf = face_centers[face_id * 3 + 2];
32
33 // Face normal and area
34 double nx = face_normals[face_id * 3 + 0];
35 double ny = face_normals[face_id * 3 + 1];
36 double nz = face_normals[face_id * 3 + 2];
37 double area = face_areas[face_id];
38
39 // r · n * A contribution
40 volume += (1.0/3.0) * (xf * nx + yf * ny + zf * nz) * area;
41 }
42 }
43
44 cell_volumes[idx] = fabs(volume); // Ensure positive volume
45}

53 int num_faces
54) {
55 int idx = blockIdx.x * blockDim.x + threadIdx.x;
56 if (idx >= num_faces) return;
57
58 // Get quad vertices
59 double x1 = face_coords[idx * 12 + 0], y1 = face_coords[idx * 12 + 1], z1 = face_coords[idx * 12 + 2];
60 double x2 = face_coords[idx * 12 + 3], y2 = face_coords[idx * 12 + 4], z2 = face_coords[idx * 12 + 5];
61 double x3 = face_coords[idx * 12 + 6], y3 = face_coords[idx * 12 + 7], z3 = face_coords[idx * 12 + 8];
62 double x4 = face_coords[idx * 12 + 9], y4 = face_coords[idx * 12 + 10], z4 = face_coords[idx * 12 + 11];
63
64 // Calculate face center
65 face_centers[idx * 3 + 0] = 0.25 * (x1 + x2 + x3 + x4);
66 face_centers[idx * 3 + 1] = 0.25 * (y1 + y2 + y3 + y4);
67 face_centers[idx * 3 + 2] = 0.25 * (z1 + z2 + z3 + z4);
68
69 // Calculate area using cross product of diagonals
70 double d1x = x3 - x1, d1y = y3 - y1, d1z = z3 - z1; // Diagonal 1
71 double d2x = x4 - x2, d2y = y4 - y2, d2z = z4 - z2; // Diagonal 2
72
73 // Cross product d1 × d2
74 double nx = d1y * d2z - d1z * d2y;
75 double ny = d1z * d2x - d1x * d2z;
76 double nz = d1x * d2y - d1y * d2x;
77
78 // Area is half the magnitude of cross product
79 double area = 0.5 * sqrt(nx*nx + ny*ny + nz*nz);
80 face_areas[idx] = area;
81
82 // Unit normal vector
83 double inv_norm = 1.0 / sqrt(nx*nx + ny*ny + nz*nz + 1e-12);
84 face_normals[idx * 3 + 0] = nx * inv_norm;
85 face_normals[idx * 3 + 1] = ny * inv_norm;
86 face_normals[idx * 3 + 2] = nz * inv_norm;
87}
