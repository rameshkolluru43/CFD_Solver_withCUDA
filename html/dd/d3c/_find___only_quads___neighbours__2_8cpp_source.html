18// Structure to store a sorted edge representation
19struct Edge {
20 int p1, p2;
21 Edge(int a, int b) {
22 if (a < b) { p1 = a; p2 = b; }
23 else { p1 = b; p2 = a; }
24 }
25 bool operator<(const Edge& other) const {
26 return (p1 < other.p1) || (p1 == other.p1 && p2 < other.p2);
27 }
28};

43 // Bounding box values

45 double x_max = numeric_limits<double>::lowest();
46 double y_min = numeric_limits<double>::max();
47 double y_max = numeric_limits<double>::lowest();
48
49 // Compute bounding box
50 for (const auto& p : points) {
51 x_min = min(x_min, p.first);
52 x_max = max(x_max, p.first);
53 y_min = min(y_min, p.second);
54 y_max = max(y_max, p.second);
55 }

57 // Count occurrences of each edge
58 for (size_t i = 0; i < cells.size(); ++i) {
59 const auto& cell = cells[i];

61 for (int j = 0; j < num_pts; ++j) {
62 Edge edge(cell[j], cell[(j + 1) % num_pts]);
63 edgeCount[edge]++;
64 }
65 }

67 // Identify boundary edges
68 set<Edge> boundaryEdges;
69 for (const auto& pair : edgeCount) {
70 if (pair.second == 1) { // Edge appears only once (boundary)
71 boundaryEdges.insert(pair.first);
72 }
73 }

75 // Identify boundary cells
76 for (size_t i = 0; i < cells.size(); ++i) {
77 const auto& cell = cells[i];
78 int num_pts = cell.size();
79 set<Edge> cellEdges;

102 boundaryType = "Left (Inlet)";
103 } else if (all_of(x_coords.begin(), x_coords.end(), [&](double x) { return fabs(x - x_max) < 1e-3; })) {

108 boundaryType = "TopWall";

111 double right_dist = x_max - *max_element(x_coords.begin(), x_coords.end());
112 double bottom_dist = *min_element(y_coords.begin(), y_coords.end()) - y_min;

116

118 else if (min_dist == fabs(right_dist)) boundaryType = "Right (Outlet)";
119 else if (min_dist == fabs(bottom_dist)) boundaryType = "BottomWall";
120 else if (min_dist == fabs(top_dist)) boundaryType = "TopWall";
121 }
