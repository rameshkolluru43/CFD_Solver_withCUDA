347void u_update()
348{
349 //------Local variables declaration--------------------------------------------------------
350 int i, j;
351 double bound = 100.0;
352 double Fe[IMAX+1][JMAX+1],Fw[IMAX+1][JMAX+1],Fn[IMAX+1][JMAX+1],Fs[IMAX+1][JMAX+1];
353 double De[IMAX+1][JMAX+1],Dw[IMAX+1][JMAX+1],Dn[IMAX+1][JMAX+1],Ds[IMAX+1][JMAX+1];
354 double Peclet_x[IMAX+1][JMAX+1],Peclet_y[IMAX+1][JMAX+1];
355 double Source[IMAX+1][JMAX+1];
356 double Previous[IMAX+1][JMAX+1];
357 //------Noting the cpu start time --------------------------------------------------------
358 double time_before = clock();
359 //=============================================== FLUX COMPUTATION =========================
360 for (j = 1; j <= JMAX-1; j++)
361 {
362 for (i = 1; i <= IMAX-2; i++)
363 {
364 //------Define cell size and distances----------
365 double dx_p = 0.5*(dx[i] + dx[i+1]);
366 double dy_p = dy[j];
367 double dx_e = dx[i+1];
368 double dx_w = dx[i];
369 double dy_n = 0.5*(dy[j] + dy[j+1]);
370 double dy_s;
371 if (j == 1) dy_s = dy[j];
372 else dy_s = 0.5*(dy[j-1] + dy[j]);
373 //------Computation of scalar values at face-centers--
374 double mue_n = 0.25*(mue(i,j) + mue(i+1,j) + mue(i,j+1) + mue(i+1,j+1));
375 double mue_s = 0.25*(mue(i,j) + mue(i+1,j) + mue(i,j-1) + mue(i+1,j-1));
376 double mue_e = mue(i+1,j);
377 double mue_w = mue(i,j);
378
379 double vf_n = 0.25*(void_frac(i,j)+void_frac(i+1,j)+void_frac(i,j+1)+void_frac(i+1,j+1));
380 double vf_s = 0.25*(void_frac(i,j)+void_frac(i+1,j)+void_frac(i,j-1)+void_frac(i+1,j-1));
381 double vf_e = void_frac(i+1,j);
382 double vf_w = void_frac(i,j);
383 if (j == 1) // Bottom Wall
384 {
385 mue_s = 0.25 * (mue(i,j) + mue(i+1,j) + mue(i,j) + mue(i+1,j));
386 vf_s = 0.25 * (void_frac(i,j) + void_frac(i+1,j) + void_frac(i,j) + void_frac(i+1,j));
387 }
388 else if (j == JMAX-1) // Top Wall
389 {
390 mue_n = 0.25 * (mue(i,j) + mue(i+1,j) + mue(i,j) + mue(i+1,j));
391 vf_n = 0.25 * (void_frac(i,j) + void_frac(i+1,j) + void_frac(i,j) + void_frac(i+1,j));
392 }
393 //------X Staggered velocity at cell faces-----
394 double U_p = u[i][j];
395 double U_pe = u[i+1][j];
396 double U_pw = u[i-1][j];
397 double U_e = 0.5*(U_p + U_pe);
398 double U_w = 0.5*(U_p + U_pw);
399 //------Y Staggered velocity at cell faces-----
400 double V_nw = v[i][j];
401 double V_ne = v[i+1][j];
402 double V_sw = v[i][j-1];
403 double V_se = v[i+1][j-1];
404 double V_n = 0.5*(V_nw + V_ne);
405 double V_s = 0.5*(V_sw + V_se);
406 double V_e = 0.5*(V_ne + V_se);
407 double V_w = 0.5*(V_nw + V_sw);
408 //------ Convective Fluxes in the X direction--
409 Fe[i][j] = (1.00/dx_p) * 0.50 * rho_g * vf_e * U_e;
410 Fw[i][j] = (1.00/dx_p) * 0.50 * rho_g * vf_w * U_w;
411 Fn[i][j] = (1.00/dy_p) * 0.50 * rho_g * vf_n * V_n;
412 Fs[i][j] = (1.00/dy_p) * 0.50 * rho_g * vf_s * V_s;
413 //------ Diffusive Fluxes in the X direction--
414 De[i][j] = (vf_e*mue_e/(dx_p*dx_e));
415 Dw[i][j] = (vf_w*mue_w/(dx_p*dx_w));
416 Dn[i][j] = (vf_n*mue_n/(dy_p*dy_n));
417 Ds[i][j] = (vf_s*mue_s/(dy_p*dy_s));
418 //------ Sources and Sinks in the X direction--
419 //================================================TURBULENT KINETIC ENERGY COMPUTATION====================
420 double KE_E = ke[i+1][j];
421 double KE_P = ke[i][j];
422 double Turbulent_KE_term= (2.0/3.0) * rho_g * (vf_e *KE_E - vf_w *KE_P)/dx_p;
423 double Fgp = 0.50 * ( F_gp(i,j) + F_gp(i+1,j) );
424 double Ext_forces = - Fgp * (u[i][j] - us[i][j]);
425 Source[i][j] = Ext_forces - Turbulent_KE_term;
426 // Determination of Peclet number
427 Peclet_x[i][j] = (U_e + U_w)* dx_p/(mue_e + mue_w);
428 Peclet_y[i][j] = (V_n + V_s)* dy_p/(mue_n + mue_s);
429 // Stores value of the previous iteration
430 Previous[i][j] = 0.00;
431 }
432 }
433 int explicit_convection_flag = 1;
434 int explicit_diffusion_flag = 0;
435/*------Loop for the determination of u* (f)---------------------------------------------*/
436 while (bound>tol)
437 {
438 Conv_U++;
439 /*------Begins thread level parallelization--------------------------------------*/
440 #pragma omp parallel private(i,j,tid)
441 {
442 /*------Domain decomposition for each thread is performed----------------*/
443 tid = omp_get_thread_num();
444 nthreads = omp_get_max_threads();
445 int start_j = 1 + tid*int(JMAX/nthreads);
446 int finish_j= start_j + JMAX/nthreads - 1;
447 if(tid == 0)
448 {
449 start_j = 1;
450 }
451 if(tid == nthreads-1)
452 {
453 finish_j = JMAX-1;
454 }
455 /*------Traversal of the grid and computation loop begins---------------*/
456 for (j = start_j; j <= finish_j; j++)
457 {
458 for (i = 0; i <= IMAX-1; i++)
459 {
460 if(i == 0) // Left Wall
461 {
462 f[0][j] = u_left;
463 }
464 else if (i == IMAX-1) // Right Wall
465 {
466 f[IMAX-1][j] = u_right;
467 }
468 else
469 {
470 /*------Computation of values at cell-center---*/
471 double vf_p = 0.50*(void_frac(i,j) + void_frac(i+1,j));
472 /*------Define the velocity variables-----------*/
473 double U_p = u[i][j];
474 double U_pn = u[i][j+1];
475 double U_ps = u[i][j-1];
476 double U_pe = u[i+1][j];
477 double U_pw = u[i-1][j];
478 /*------Define the intermediate velocity variables-----------*/
479 double f_p = f[i][j];
480 double f_pn = f[i][j+1];
481 double f_ps = f[i][j-1];
482 double f_pe = f[i+1][j];
483 double f_pw = f[i-1][j];
484 //------Adjustment of values for boundaries--
485 if (j == 1) // Bottom Wall
486 {
487 f_ps = 2.00 * u_bottom - f[i][j];
488 }
489 else if (j == JMAX-1) // Top Wall
490 {
491 f_pn = 2.00 * u_top - f[i][j];
492 }
493 //=============================================== CONVECTION COMPUTATION ========================================//
494 //------Convection term(Implicit)
495 double DUUDX, DUVDY;
496 double alpha_x= 1.00;
497 if (Peclet_x[i][j] < - 2) alpha_x = -1;
498 else if (Peclet_x[i][j] > 2) alpha_x = 1;
499 else alpha_x = 0;
500 double alpha_y= 1.00;
501 if (Peclet_y[i][j] < - 2) alpha_y = -1;
502 else if (Peclet_y[i][j] > 2) alpha_y = 1;
503 else alpha_y = 0;
504 if( explicit_convection_flag == 1)
505 {
506 DUUDX = ( Fe[i][j] * ((U_p + U_pe) + alpha_x * (U_p - U_pe)) - Fw[i][j] * ((U_p + U_pw) + alpha_x * (U_pw - U_p)) );
507 DUVDY = ( Fn[i][j] * ((U_p + U_pn) + alpha_y * (U_p - U_pn)) - Fs[i][j] * ((U_p + U_ps) + alpha_y * (U_ps - U_p)) );
508 }
509 else
510 {
511 DUUDX = ( Fe[i][j] * ((f_p + f_pe) + alpha_x * (f_p - f_pe)) - Fw[i][j] * ((f_p + f_pw) + alpha_x * (f_pw - f_p)) );
512 DUVDY = ( Fn[i][j] * ((f_p + f_pn) + alpha_y * (f_p - f_pn)) - Fs[i][j] * ((f_p + f_ps) + alpha_y * (f_ps - f_p)) );
513 }
514 double Convec_term = DUUDX + DUVDY;
515 //================================================DIFFUSION COMPUTATION ========================================//
516 //------Diffusion term(Implicit)
517 double D2UDX2, D2UDY2;
518 if( explicit_diffusion_flag == 1)
519 {
520 D2UDX2 = (Dw[i][j] * U_pw) + (De[i][j] * U_pe);
521 D2UDY2 = (Ds[i][j] * U_ps) + (Dn[i][j] * U_pn);
522 }
523 else
524 {
525 D2UDX2 = (Dw[i][j] * f_pw) + (De[i][j] * f_pe);
526 D2UDY2 = (Ds[i][j] * f_ps) + (Dn[i][j] * f_pn);
527 }
528 double Diffusive_term = D2UDX2 + D2UDY2;
529 //================================================SOURCE COMPUTATION =========================================//
530 double Source_term = Source[i][j];
531 //================================================PREVIOUS TIME TERM COMPUTATION ============================//
532 double Previous_Time_term = (vf_p*rho_g/dt1) * U_p;
533 //================================================COMPUTATION OF DENOMINATOR ================================//
534 double aP;
535 if( ( explicit_convection_flag == 1) && ( explicit_diffusion_flag == 1) )
536 {
537 aP = (vf_p*rho_g/dt1);
538 }
539 else if( ( explicit_convection_flag == 1) && ( explicit_diffusion_flag == 0) )
540 {
541 aP = (vf_p*rho_g/dt1) + (De[i][j] + Dw[i][j] + Dn[i][j] + Ds[i][j]);
542 }
543 else if( ( explicit_convection_flag == 0) && ( explicit_diffusion_flag == 1) )
544 {
545 aP = (vf_p*rho_g/dt1) + (1.00 + alpha_x)*Fe[i][j] - (1.00 - alpha_x)*Fw[i][j] + (1.00 + alpha_y)*Fn[i][j] - (1.00 - alpha_y)*Fs[i][j];
546 }
547 else
548 {
549 aP = (vf_p*rho_g/dt1) + (1.00 + alpha_x)*Fe[i][j] - (1.00 - alpha_x)*Fw[i][j] + (1.00 + alpha_y)*Fn[i][j] - (1.00 - alpha_y)*Fs[i][j] + (De[i][j] + Dw[i][j] + Dn[i][j] + Ds[i][j]);
550 }
551 //================================================COMPUTATION of u* =======================================//
552 if (aP == 0)
553 {
554 f[i][j] = 0.0; // If the grid point is in the solid phase
555 }
556 else
557 {
558 f[i][j] = (1.00 - omega_u) * f_p + (omega_u/aP)*(Previous_Time_term - Convec_term + Diffusive_term + Source_term);
559 }
560 }
561 }
562 }
563 }

565 #pragma omp barrier
566 /*------Enforce a state-based boundary condition-----------------------------*/
567 enforce_domain_physics();
568 /*------Computation of residue to determine convergence----------------------*/

580 /*------Convergence met. u* has been determined--------------------------------------*/
581 /*------Compute the maximum value of u* for monitoring and debugging-----------------*/
582 max_Ug = 0.00;

584 {
585 for (j = 0;j <= JMAX; j++)

594 double time_after = clock();

603void v_update()
604{
605 //------Local variables declaration--------------------------------------------------------
606 int i, j;
607 double bound = 100.0;
608 double Fe[IMAX+1][JMAX+1],Fw[IMAX+1][JMAX+1],Fn[IMAX+1][JMAX+1],Fs[IMAX+1][JMAX+1];
609 double De[IMAX+1][JMAX+1],Dw[IMAX+1][JMAX+1],Dn[IMAX+1][JMAX+1],Ds[IMAX+1][JMAX+1];
610 double Peclet_x[IMAX+1][JMAX+1],Peclet_y[IMAX+1][JMAX+1];
611 double Source[IMAX+1][JMAX+1];
612 double Previous[IMAX+1][JMAX+1];
613 //------Noting the cpu start time --------------------------------------------------------
614 double time_before = clock();
615 //=============================================== FLUX COMPUTATION ==============================================
616 for (j = 1; j <= JMAX-2; j++)
617 {
618 for (i = 1; i <= IMAX-1; i++)
619 {
620 //------Define cell size and distances----------
621 double dx_p = dx[i];
622 double dy_p = 0.5*(dy[j] + dy[j+1]);
623 double dx_e = 0.5*(dx[i] + dx[i+1]);
624 double dx_w;
625 if (i == 1) dx_w = dx[i];
626 else dx_w = 0.5*(dx[i-1] + dx[i]);
627 double dy_n = dy[j+1];
628 double dy_s = dy[j] ;
629 //------Computation of values at face-centers--
630 double mue_n = mue(i,j+1);
631 double mue_s = mue(i,j);
632 double mue_e = 0.25*(mue(i,j)+mue(i+1,j)+mue(i,j+1)+mue(i+1,j+1));
633 double mue_w = 0.25*(mue(i,j)+mue(i-1,j)+mue(i,j+1)+mue(i-1,j+1));
634
635 double vf_n = void_frac(i,j+1);
636 double vf_s = void_frac(i,j);
637 double vf_e = 0.25*(void_frac(i,j)+void_frac(i+1,j)+void_frac(i,j+1)+void_frac(i+1,j+1));
638 double vf_w = 0.25*(void_frac(i,j)+void_frac(i-1,j)+void_frac(i,j+1)+void_frac(i-1,j+1));
639 //------Adjustment of values for boundaries--
640 if (i == 1) // Left Wall
641 {
642 mue_w = 0.25 * (mue(i,j)+mue(i,j)+mue(i,j+1)+mue(i,j+1));
643 vf_w = 0.25 * (void_frac(i,j)+void_frac(i,j)+void_frac(i,j+1)+void_frac(i,j+1));
644 }
645 else if(i == IMAX-1) // Right Wall
646 {
647 mue_e = 0.25 * (mue(i,j) + mue(i,j) + mue(i,j+1) + mue(i,j+1));
648 vf_e = 0.25 * (void_frac(i,j) + void_frac(i,j) + void_frac(i,j+1) + void_frac(i,j+1));
649 }
650 //------Y Staggered velocity at cell faces-----
651 double V_p = v[i][j];
652 double V_pn = v[i][j+1];
653 double V_ps = v[i][j-1];
654 double V_n = 0.5*(V_p + V_pn);
655 double V_s = 0.5*(V_p + V_ps);
656 //------X Staggered velocity at cell faces-----
657 double U_nw = u[i-1][j+1];
658 double U_ne = u[i][j+1];
659 double U_sw = u[i-1][j];
660 double U_se = u[i][j];
661 double U_n = 0.5*(U_nw + U_ne);
662 double U_s = 0.5*(U_sw + U_se);
663 double U_e = 0.5*(U_ne + U_se);
664 double U_w = 0.5*(U_nw + U_sw);
665 //------ Convective Fluxes in the Y direction--
666 Fe[i][j] = (1.00/dx_p) * 0.50 * rho_g * vf_e * U_e;
667 Fw[i][j] = (1.00/dx_p) * 0.50 * rho_g * vf_w * U_w;
668 Fn[i][j] = (1.00/dy_p) * 0.50 * rho_g * vf_n * V_n;
669 Fs[i][j] = (1.00/dy_p) * 0.50 * rho_g * vf_s * V_s;
670 //------ Diffusive Fluxes in the Y direction--
671 De[i][j] = (vf_e*mue_e/(dx_p*dx_e));
672 Dw[i][j] = (vf_w*mue_w/(dx_p*dx_w));
673 Dn[i][j] = (vf_n*mue_n/(dy_p*dy_n));
674 Ds[i][j] = (vf_s*mue_s/(dy_p*dy_s));
675 //------ Sources and Sinks in the Y direction--
676 //================================================TURBULENT KINETIC ENERGY COMPUTATION====================//
677 double KE_N = ke[i][j+1];
678 double KE_P = ke[i][j];
679 double Turbulent_KE_term= (2.0/3.0) * rho_g * (vf_n * KE_N - vf_s * KE_P)/dy_p;
680 double Fgp = 0.50 * ( F_gp(i,j) + F_gp(i,j+1) );
681 double Ext_forces = - Fgp * (v[i][j] - vs[i][j]);
682 Source[i][j] = Ext_forces - Turbulent_KE_term;
683 // Determination of Peclet number
684 Peclet_x[i][j] = (U_e + U_w)* dx_p/(mue_e + mue_w);
685 Peclet_y[i][j] = (V_n + V_s)* dy_p/(mue_n + mue_s);
686 // Stores value of the previous iteration
687 Previous[i][j] = 0.00;
688 }
689 }
690 int explicit_convection_flag = 0;
691 int explicit_diffusion_flag = 0;
692 /*------Loop for the determination of v* (g)---------------------------------------------*/
693 while (bound>tol)
694 {
695 Conv_V++;
696 /*------Begins thread level parallelization--------------------------------------*/
697 #pragma omp parallel private(i,j,tid)
698 {
699 /*------Domain decomposition for each thread is performed----------------*/
700 tid = omp_get_thread_num();
701 nthreads = omp_get_max_threads();
702 int start_j = 1 + tid*int(JMAX/nthreads);
703 int finish_j= start_j + JMAX/nthreads - 1;
704 if(tid == 0)
705 {
706 start_j = 0;
707 }
708 if(tid == nthreads-1)
709 {
710 finish_j = JMAX-1;
711 }
712 /*------Traversal of the grid and computation loop begins---------------*/
713 for (j = start_j; j <= finish_j; j++)
714 {
715 for (i = 1; i <= IMAX-1; i++)
716 {
717 if(j == 0) // Bottom Wall
718 {
719 g[i][0] = v_bottom;
720 }
721 else if(j == JMAX-1) // Top Wall
722 {
723 g[i][JMAX-1] = g[i][JMAX-2];
724 }
725 else
726 {
727 //------Computation of values at cell-center---
728 double vf_p = 0.50*(void_frac(i,j+1) + void_frac(i,j));
729 //------Define the velocity variables----------
730 double V_p = v[i][j];
731 double V_pn = v[i][j+1];
732 double V_ps = v[i][j-1];
733 double V_pe = v[i+1][j];
734 double V_pw = v[i-1][j];
735 //------Define the intermediate velocity variables-----------
736 double g_p = g[i][j];
737 double g_pn = g[i][j+1];
738 double g_ps = g[i][j-1];
739 double g_pe = g[i+1][j];
740 double g_pw = g[i-1][j];
741 //------Adjustment of values for boundaries--
742 if (i == 1) // Left Wall
743 {
744 g_pw = 2.00 * v_left - g[i][j];
745 }
746 else if(i == IMAX-1) // Right Wall
747 {
748 g_pe = 2.00 * v_right - g[i][j];
749 }
750 //=============================================== CONVECTION COMPUTATION ========================================//
751 //------Convection term(Implicit)
752 double DUVDX, DVVDY;
753 double alpha_x = 1.00;
754 if (Peclet_x[i][j] < - 2) alpha_x = -1;
755 else if (Peclet_x[i][j] > 2) alpha_x = 1;
756 else alpha_x = 0;
757 double alpha_y = 1.00;
758 if (Peclet_y[i][j] < - 2) alpha_y = -1;
759 else if (Peclet_y[i][j] > 2) alpha_y = 1;
760 else alpha_y = 0;
761 if( explicit_convection_flag == 1)
762 {
763 DUVDX = ( Fe[i][j] * ((V_p + V_pe) + alpha_x * (V_p - V_pe)) - Fw[i][j] * ((V_pw + V_p) + alpha_x * (V_pw - V_p)) );
764 DVVDY = ( Fn[i][j] * ((V_p + V_pn) + alpha_y * (V_p - V_pn)) - Fs[i][j] * ((V_ps + V_p) + alpha_y * (V_ps - V_p)) );
765 }
766 else
767 {
768 DUVDX = ( Fe[i][j] * ((g_p + g_pe) + alpha_x * (g_p - g_pe)) - Fw[i][j] * ((g_pw + g_p) + alpha_x * (g_pw - g_p)) );
769 DVVDY = ( Fn[i][j] * ((g_p + g_pn) + alpha_y * (g_p - g_pn)) - Fs[i][j] * ((g_ps + g_p) + alpha_y * (g_ps - g_p)) );
770 }
771 double Convec_term = DUVDX + DVVDY;
772 //================================================DIFFUSION COMPUTATION ========================================//
773 /*------Diffusion term(Implicit for stability)*/
774 double D2VDX2, D2VDY2;
775 if( explicit_diffusion_flag == 1)
776 {
777 D2VDX2 = (Dw[i][j] * V_pw) + (De[i][j] * V_pe);
778 D2VDY2 = (Ds[i][j] * V_ps) + (Dn[i][j] * V_pn);
779 }
780 else
781 {
782 D2VDX2 = (Dw[i][j] * g_pw) + (De[i][j] * g_pe);
783 D2VDY2 = (Ds[i][j] * g_ps) + (Dn[i][j] * g_pn);
784 }
785 double Diffusive_term = D2VDX2 + D2VDY2;
786 //================================================SOURCE COMPUTATION =====================================//
787 double Source_term = Source[i][j];
788 /*------Previous time term computation--------*/
789 double Previous_Time_term = (vf_p*rho_g/dt1) * V_p;
790 /*------Computation of v*---------------------*/
791 double aP;
792 if( ( explicit_convection_flag == 1) && ( explicit_diffusion_flag == 1) )
793 {
794 aP = (vf_p*rho_g/dt1);
795 }
796 else if( ( explicit_convection_flag == 1) && ( explicit_diffusion_flag == 0) )
797 {
798 aP = (vf_p*rho_g/dt1) + (De[i][j] + Dw[i][j] + Dn[i][j] + Ds[i][j]);
799 }
800 else if( ( explicit_convection_flag == 0) && ( explicit_diffusion_flag == 1) )
801 {
802 aP = (vf_p*rho_g/dt1) + (1.00 + alpha_x)*Fe[i][j] - (1.00 - alpha_x)*Fw[i][j] + (1.00 + alpha_y)*Fn[i][j] - (1.00 - alpha_y)*Fs[i][j];
803 }
804 else
805 {
806 aP = (vf_p*rho_g/dt1) + (1.00 + alpha_x)*Fe[i][j] - (1.00 - alpha_x)*Fw[i][j] + (1.00 + alpha_y)*Fn[i][j] - (1.00 - alpha_y)*Fs[i][j] + (De[i][j] + Dw[i][j] + Dn[i][j] + Ds[i][j]);
807 }
808 if (aP == 0)
809 {
810 g[i][j] = 0.0; // If the grid point is in the solid phase
811 }
812 else
813 {
814 g[i][j] = (1.00 - omega_v) * g_p + (omega_v/aP)*(Previous_Time_term - Convec_term + Diffusive_term + Source_term);
815 }
816 }
817 }
818 }
819 }
820 /*------Thread synchronization is performed at this point--------------------*/
821 #pragma omp barrier
822 /*------Enforce a state-based boundary condition-----------------------------*/
823 enforce_domain_physics();
824 /*------Computation of residue to determine convergence----------------------*/
825 bound = 0.0;
826 for (i = 1; i <= IMAX-1; i++)
827 {
828 for (j = 0; j <= JMAX-1; j++)
829 {
830 bound = bound + fabs(g[i][j] - Previous[i][j]);
831 Previous[i][j] = g[i][j];
832 }
833 }
834 if( ( explicit_convection_flag == 1) && ( explicit_diffusion_flag == 1) ) bound = -tol;
835 }
836 /*------Convergence met. v* has been determined--------------------------------------*/
837 /*------Compute the maximum value of v* for monitoring and debugging-----------------*/
838 max_Vg = 0.00;
839 for (i = 0; i <= IMAX; i++)
840 {
841 for (j = 0; j <= JMAX; j++)
842 {
843 if (abs(g[i][j]) > max_Vg)
844 {
845 max_Vg = abs(g[i][j]);
846 }
847 }
848 }
849 /*------Noting the cpu end time ------------------------------------------------------*/
850 double time_after = clock();
851 /*------Enforce a state-based boundary condition--------------------------------------*/
852 enforce_domain_physics();
853 /*------Noting the total cpu time for this computation--------------------------------*/
854 cpu_time_V_gas = ((double) (time_after - time_before)) / (nthreads*CLOCKS_PER_SEC);
855}
856//=============================================================================================
857// GAS PRESSURE EQUATION SOLUTION
858//=============================================================================================
859void p_update()
860{
861 /*------Local variables declaration--------------------------------------------------------*/
862 int i, j;
863 double bound = 100.0;
864 double bound_old = 100.0;
865 double Previous[IMAX+1][JMAX+1];
866 int loop = 0;
867 /*------Noting the cpu start time ---------------------------------------------------------*/
868 double time_before = clock();
869 /*------Initialization the variable that stores the previous iteration value---------------*/
870 for (i = 1; i <= IMAX-1; i++)
871 {
872 for (j=1; j<=JMAX-1; j++)
873 {
874 Previous[i][j] = p[i][j];
875 }
876 }
877 /*------Loop for the determination of p----------------------------------------------------*/
878 while (bound > tol)
879 {
880 Conv_P++;
881 /*------Begins thread level parallelization---------------------------------------*/
882 #pragma omp parallel private(i,j,tid)
883 {
884 /*------Private variables for each thread is declared---------------------*/
885 double DUDX, DVDY;
886 double D2PDX2, D2PDY2;
887 double RHS, coefficient;
888 double dx_p, dx_e, dx_w, dy_p, dy_n ,dy_s;
889 double P_P, P_N, P_S, P_W, P_E;
890 double f_w, f_e, g_n, g_s;
891 /*------Domain decomposition for each thread is performed-----------------------------*/
892 tid = omp_get_thread_num();
893 nthreads = omp_get_max_threads();
894 int start_j = 1 + tid*int(JMAX/nthreads);
895 int finish_j= start_j + JMAX/nthreads - 1;
896 if(tid == 0)
897 {
898 start_j = 1;
899 }
900 if(tid == nthreads-1)
901 {
902 finish_j = JMAX-1;
903 }
904 /*------Traversal of the grid and computation loop begins-----------------------------*/
905 for (i = 1; i <= IMAX-1; i++)
906 {
907 for (j = start_j; j <= finish_j; j++)
908 {
909 /*------Define cell size and distances--------------------------------*/
910 dx_p=dx[i];
911 dy_p=dy[j];
912 if (i==IMAX-1) dx_e=dx[i];
913 else dx_e=0.5*(dx[i]+dx[i+1]);
914
915 if (i==1) dx_w=dx[i];
916 else dx_w=0.5*(dx[i-1]+dx[i]);
917
918 if (j==JMAX-1) dy_n=dy[j];
919 else dy_n=0.5*(dy[j]+dy[j+1]);
920
921 if (j==1) dy_s=dy[j];
922 else dy_s=0.5*(dy[j-1]+dy[j]);
923 /*------Define the approximate gas velocity variables-----------------*/
924 f_w = f[i-1][j];
925 f_e = f[i][j];
926 g_n = g[i][j];
927 g_s = g[i][j-1];
928 /*------Define the variables with gas pressure at previous timestep---*/
929 P_P = Previous[i][j];
930 P_N = p[i][j+1];
931 P_S = p[i][j-1];
932 P_W = p[i-1][j];
933 P_E = p[i+1][j];
934 /*------Adjustment of values for boundaries--------------------------*/
935 if(i == 1) // Left Wall
936 {
937 P_W = p[i][j];
938 if(j == 1) // Left-Bottom Corner
939 {
940 P_S = p[i][j];
941 }
942 else if(j == JMAX-1) // Left-Top Corner
943 {
944 P_N = 0;
945 }
946 }
947 else if(i == IMAX-1) // Right Wall
948 {
949 P_E = p[i][j];
950 if(j == 1) // Right-Bottom Corner
951 {
952 P_S = p[i][j];
953 }
954 else if(j == JMAX-1) // Right-Top Corner
955 {
956 P_N = 0;
957 }
958 }
959 else
960 {
961 if(j == 1) // Bottom Wall
962 {
963 P_S = p[i][j];
964 }
965 else if(j == JMAX-1) // Top Wall
966 {
967 P_N = 0;
968 }
969 }
970 /*------Coefficient term computation---------------------------------*/
971 double aE= (1.0/(dx_p*dx_e));
972 double aW= (1.0/(dx_p*dx_w));
973 double aN= (1.0/(dy_p*dy_n));
974 double aS= (1.0/(dy_p*dy_s));
975 double aP= (aE + aW + aN + aS);
976 /*------Computation of pressure derivative terms---------------------*/
977 D2PDX2 = aW*P_W + aE*P_E;
978 D2PDY2 = aS*P_S + aN*P_N;
979 /*------Computation of divergence of approximate velocity(RHS)-------*/
980 DUDX = (f_e - f_w)/dx_p;
981 DVDY = (g_n - g_s)/dy_p;
982 RHS = (rho_g/dt1)*(DUDX + DVDY);
983 /*------Computation of pressure term---------------------------------*/
984 p[i][j] = ((1.0 - omega_p) * P_P + (omega_p/aP) * (D2PDX2 + D2PDY2 - RHS));
985 }
986 }
987 }
988 /*------Thread synchronization is performed at this point------------------------------------*/
989 #pragma omp barrier
990 /*------Computation of residue to determine convergence--------------------------------------*/
991 bound = 0.00;
992 max_P = 0.00;
993 for (i = 1; i <= IMAX-1; i++)
994 {
995 for (j = 1; j <= JMAX-1; j++)
996 {
997 bound = bound + fabs( p[i][j] - Previous[i][j] );
998 Previous[i][j] = relaxp * p[i][j];
999 if (p[i][j] > max_P)
1000 {
1001 max_P = p[i][j];
1002 }
1003 }
1004 }
1005
1006 double r = float(rand())/float(RAND_MAX);
1007 if (bound > bound_old)
1008 {
1009 omega_p = 1.00 - r * 0.50;
1010 }
1011 else
1012 {
1013 omega_p = 1.00 + r * 0.50;
1014 }
1015 bound_old = bound;
1016 loop = loop + 1;
1017 if (loop == 1) printf("Pressure Evolution :: Iteration\t Bound\t\tMax P\t Int.Relaxation\tExt.Relaxation\n");
1018 if ((loop%10000) == 0) printf("Pressure Evolution :: %d\t%lf\t %lf\t%lf\t%lf\n", loop, bound, max_P, omega_p, relaxp);
1019
1020 if (loop == 100000)
1021 {
1022 printf("Pressure Evolution :: Forced Exit \n");
1023 break;
1024 }
1025 }
1026 /*------Convergence met. p has been determined------------------------------------------------------*/
1027 /*------Noting the cpu end time ----------------------------------------------------------------------*/
1028 double time_after = clock();
1029 /*------Enforce a state-based boundary condition------------------------------------------------------*/
1030 enforce_domain_physics();
1031 /*------Noting the total cpu time for this computation------------------------------------------------*/
1032 cpu_time_Pressure = ((double) (time_after - time_before)) / (nthreads*CLOCKS_PER_SEC);
1033}

1094 v[i][0] = v_bottom;
1095 v[i][JMAX-1] = v[i][JMAX-2];
1096 }
