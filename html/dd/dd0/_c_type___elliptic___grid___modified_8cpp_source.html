21void initializeGrid() {
22 // Boundary conditions for the outer rectangular boundary
23 for (int i = 0; i < NX; ++i) {
24 double xi = (double)i / (NX - 1);
25 x[i][0] = -Lx + 2 * Lx * xi; // Bottom boundary (y = -Ly)
26 y[i][0] = -Ly;
27
28 x[i][NY - 1] = -Lx + 2 * Lx * xi; // Top boundary (y = +Ly)
29 y[i][NY - 1] = Ly;
30 }
31
32 for (int j = 0; j < NY; ++j) {
33 double eta = (double)j / (NY - 1);
34 x[0][j] = -Lx; // Left boundary (x = -Lx)
35 y[0][j] = -Ly + 2 * Ly * eta;
36
37 x[NX - 1][j] = Lx; // Right boundary (x = +Lx)
38 y[NX - 1][j] = -Ly + 2 * Ly * eta;
39 }
40
41 // Boundary conditions for the cylinder
42 int numCylinderPoints = NX - 2; // Number of points to place around the cylinder
43 for (int i = 0; i < numCylinderPoints; ++i) {
44 double theta = 2.0 * M_PI * i / numCylinderPoints; // Theta from 0 to 2π
45 x[i][1] = cylinderCenterX + cylinderRadius * cos(theta); // Circle around cylinder
46 y[i][1] = cylinderCenterY + cylinderRadius * sin(theta);
47 }
48
49 // Ensure closure of the cylinder by duplicating the first point at the end
50 x[numCylinderPoints][1] = x[0][1];
51 y[numCylinderPoints][1] = y[0][1];
52
53 // Initial guess for internal points (linear interpolation)
54 for (int i = 1; i < NX - 1; ++i) {
55 for (int j = 1; j < NY - 1; ++j) {
56 double xi = (double)i / (NX - 1);
57 double eta = (double)j / (NY - 1);
58 x[i][j] = (1 - eta) * x[i][0] + eta * x[i][NY - 1]; // Interpolate vertically
59 y[i][j] = (1 - eta) * y[i][0] + eta * y[i][NY - 1];
60 }

64// Laplace smoothing solver to adjust interior points
65void solveEllipticGrid() {
66 double maxError;
67 int iteration = 0;
68
69 do {
70 maxError = 0.0;
71
72 for (int i = 1; i < NX - 1; ++i) {
73 for (int j = 2; j < NY - 2; ++j) { // Skip the cylinder boundary at j = 1
74 double x_old = x[i][j];
75 double y_old = y[i][j];
76
77 // Laplace equation for smoothing (interior points only)
78 x[i][j] = 0.25 * (x[i + 1][j] + x[i - 1][j] + x[i][j + 1] + x[i][j - 1]);
79 y[i][j] = 0.25 * (y[i + 1][j] + y[i - 1][j] + y[i][j + 1] + y[i][j - 1]);
80
81 double error = std::max(std::abs(x[i][j] - x_old), std::abs(y[i][j] - y_old));
82 maxError = std::max(maxError, error);
83 }
84 }
85
86 // Increment iteration count and print progress every 100 iterations
87 iteration++;
88 if (iteration % 100 == 0) {
89 std::cout << "Iteration: " << iteration << ", Max Error: " << maxError << std::endl;
90 }
91
92 } while (maxError > tolerance && iteration < maxIterations);
93
94 std::cout << "Final iteration: " << iteration << ", Final Max Error: " << maxError << std::endl;
95}
