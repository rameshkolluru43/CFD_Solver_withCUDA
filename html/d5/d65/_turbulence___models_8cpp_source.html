182void Calculate_Strain_Rate_Tensor(int cell_index, double S[3][3])
183{
184 // Initialize tensor
185 for (int i = 0; i < 3; i++)
186 {
187 for (int j = 0; j < 3; j++)
188 {
189 S[i][j] = 0.0;
190 }
191 }
192
193 if (Cells[cell_index].cellType == GHOST_CELL)
194 return;
195
196 // Get velocity gradients from cell
197 Vector u_grad = Cells[cell_index].Get_Grad_UatCenter();
198 Vector v_grad = Cells[cell_index].Get_Grad_VatCenter();
199 Vector w_grad = Cells[cell_index].Get_Grad_WatCenter();
200
201 // Calculate strain rate tensor: Sij = 0.5 * (∂ui/∂xj + ∂uj/∂xi)
202 S[0][0] = u_grad.Get_Component(1); // ∂u/∂x
203 S[1][1] = v_grad.Get_Component(2); // ∂v/∂y
204 S[2][2] = w_grad.Get_Component(3); // ∂w/∂z
205
206 S[0][1] = S[1][0] = 0.5 * (u_grad.Get_Component(2) + v_grad.Get_Component(1)); // 0.5*(∂u/∂y + ∂v/∂x)
207 S[0][2] = S[2][0] = 0.5 * (u_grad.Get_Component(3) + w_grad.Get_Component(1)); // 0.5*(∂u/∂z + ∂w/∂x)
208 S[1][2] = S[2][1] = 0.5 * (v_grad.Get_Component(3) + w_grad.Get_Component(2)); // 0.5*(∂v/∂z + ∂w/∂y)
209}

216void Calculate_Vorticity_Tensor(int cell_index, double Omega[3][3])
217{
218 // Initialize tensor
219 for (int i = 0; i < 3; i++)
220 {
221 for (int j = 0; j < 3; j++)
222 {
223 Omega[i][j] = 0.0;
224 }
225 }
226
227 if (Cells[cell_index].cellType == GHOST_CELL)
228 return;
229
230 // Get velocity gradients from cell
231 Vector u_grad = Cells[cell_index].Get_Grad_UatCenter();
232 Vector v_grad = Cells[cell_index].Get_Grad_VatCenter();
233 Vector w_grad = Cells[cell_index].Get_Grad_WatCenter();
234
235 // Calculate vorticity tensor: Ωij = 0.5 * (∂ui/∂xj - ∂uj/∂xi)
236 Omega[0][1] = 0.5 * (u_grad.Get_Component(2) - v_grad.Get_Component(1)); // 0.5*(∂u/∂y - ∂v/∂x)
237 Omega[1][0] = -Omega[0][1];
238
239 Omega[0][2] = 0.5 * (u_grad.Get_Component(3) - w_grad.Get_Component(1)); // 0.5*(∂u/∂z - ∂w/∂x)
240 Omega[2][0] = -Omega[0][2];
241
242 Omega[1][2] = 0.5 * (v_grad.Get_Component(3) - w_grad.Get_Component(2)); // 0.5*(∂v/∂z - ∂w/∂y)
243 Omega[2][1] = -Omega[1][2];
244}

308void Calculate_Wall_Shear_Stress(int cell_index)
309{
310 if (Cells[cell_index].cellType == GHOST_CELL || !Cells[cell_index].hasBoundaryface)
311 return;
312
313 double rho = Cells[cell_index].Conservative_Variables[0];
314 double u = Cells[cell_index].Conservative_Variables[1] / rho;
315 double v = Cells[cell_index].Conservative_Variables[2] / rho;
316 double w = Cells[cell_index].Conservative_Variables[3] / rho;
317
318 double U_magnitude = sqrt(u * u + v * v + w * w);
319 double mu = Calculate_Laminar_Viscosity(cell_index);
320 double y_wall = turbulence_vars[cell_index].y_wall;
321
322 // Simple wall shear stress calculation
323 if (y_wall > 0)
324 {
325 turbulence_vars[cell_index].tau_wall = mu * U_magnitude / y_wall;
326 }
327 else
328 {
329 turbulence_vars[cell_index].tau_wall = 0.0;
330 }

337void Calculate_Y_Plus(int cell_index)
338{
339 if (Cells[cell_index].cellType == GHOST_CELL)
340 return;
341
342 double rho = Cells[cell_index].Conservative_Variables[0];
343 double tau_wall = turbulence_vars[cell_index].tau_wall;
344 double mu = Calculate_Laminar_Viscosity(cell_index);
345 double y_wall = turbulence_vars[cell_index].y_wall;
346
347 if (tau_wall > 0)
348 {
349 turbulence_vars[cell_index].u_tau = sqrt(tau_wall / rho);
350 turbulence_vars[cell_index].y_plus = rho * turbulence_vars[cell_index].u_tau * y_wall / mu;
351 }
352 else
353 {
354 turbulence_vars[cell_index].u_tau = 0.0;
355 turbulence_vars[cell_index].y_plus = 0.0;
