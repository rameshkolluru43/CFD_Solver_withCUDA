272static std::vector<double> readCsv1D(const std::string &path)
273{
274 std::ifstream f(path);
275 if (!f.is_open())
276 {
277 std::cerr << "Failed to open CSV: " << path << "\n";
278 exit(1);
279 }
280 std::vector<double> vals;
281 std::string line;
282 while (std::getline(f, line))
283 {
284 if (line.empty())
285 continue;
286 std::stringstream ss(line);
287 std::string tok;
288 while (std::getline(ss, tok, ','))
289 {
290 if (tok.empty())
291 continue;
292 // allow whitespace separated values too
293 std::stringstream ts(tok);
294 double v;
295 if ((ts >> v))
296 vals.push_back(v);
297 }
298 }
299 if (vals.empty())
300 {
301 std::cerr << "No numeric values found in CSV: " << path << "\n";
302 exit(1);
303 }
304 return vals;
305}
