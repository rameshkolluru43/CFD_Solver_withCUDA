144inline void cudaCheckKernel(const char *kernel_name)
145{
146 cudaError_t err = cudaGetLastError();
147 if (err != cudaSuccess)
148 {
149 printf("CUDA Kernel Error in %s: %s\n", kernel_name, cudaGetErrorString(err));
150 exit(1);
151 }
152}
