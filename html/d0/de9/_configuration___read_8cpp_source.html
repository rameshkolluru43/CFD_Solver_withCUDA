81// Function to read parameters from a configuration file
82map<string, string> ReadConfigFile(const string &filename)
83{
84 map<string, string> config;
85 ifstream file(filename);
86
87 if (!file.is_open())
88 {
89 cerr << "Error: Unable to open configuration file " << filename << endl;
90 exit(EXIT_FAILURE);
91 }
92
93 string line;
94 while (getline(file, line))
95 {
96 if (line.empty() || line[0] == '#')
97 continue; // Skip empty lines and comments
98 size_t delimiter = line.find('=');
99 if (delimiter == string::npos)
100 continue;
101
102 string key = line.substr(0, delimiter);
103 string value = line.substr(delimiter + 1);
104
105 // Trim whitespace from key and value
106 key.erase(key.find_last_not_of(" \t") + 1);
107 value.erase(0, value.find_first_not_of(" \t"));
108 value.erase(value.find_last_not_of(" \t") + 1);
109
110 config[key] = value;
111 }
112 file.close();
113 return config;
114}
