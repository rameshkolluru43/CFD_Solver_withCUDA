328void mapFacesToCells(vector<Cell> &cells, map<pair<int, int>, set<int>> &faceToCells)
329{
330 bool restartNeeded = false;
331 set<pair<int, int>> nonManifoldFaces; // Store faces that are shared by more than two cells
332
333 for (size_t cellID = 0; cellID < cells.size(); ++cellID)
334 {
335 const auto &cellNodes = cells[cellID].nodeIndices;
336 // cout << "Cell ID: " << cellID << "\t";
337 // For each cell go through its nodes and create a faces
338 for (size_t i = 0; i < cellNodes.size(); ++i)
339 {
340 int node1 = cellNodes[i];
341 int node2 = cellNodes[(i + 1) % cellNodes.size()];
342
343 // Ensure consistent ordering of face
344 if (node1 > node2)
345 swap(node1, node2);
346 pair<int, int> face = {node1, node2};
347
348 // Print face nodes
349 // cout << "Face Nodes: " << node1 << "\t" << node2 << "\t";
350 // search face in the list of faces
351 auto it = faceToCells.find(face);
352 if (it != faceToCells.end())
353 {
354 // The face is already stored, meaning it's shared
355 set<int> &sharedCells = it->second;
356 sharedCells.insert(cellID);
357
358 if (sharedCells.size() == 2)
359 {
360 // The face is now shared between exactly two cells (valid)
361 auto it = sharedCells.begin();
362 int firstCell = *it++;
363 int secondCell = *it;
364
365 // ✅ Use push_back() instead of overwriting Neighbours[i]
366 if (find(cells[firstCell].Neighbours.begin(), cells[firstCell].Neighbours.end(), secondCell) == cells[firstCell].Neighbours.end())

368 cells[firstCell].Neighbours.push_back(secondCell);

370 if (find(cells[secondCell].Neighbours.begin(), cells[secondCell].Neighbours.end(), firstCell) == cells[secondCell].Neighbours.end())

372 cells[secondCell].Neighbours.push_back(firstCell);
373 }

642// Helper: Given a face's normal (nx, ny), returns a priority integer:
643// 1 -> left, 2 -> bottom, 3 -> right, 4 -> top.
644// The assignment is made by comparing the normal's angle with the target angles:
645// left: π (or –π), bottom: -π/2, right: 0, top: π/2.
646int compute_face_priority(double nx, double ny)
647{
648 // Compute the angle in radians; range is (-PI, PI]
649 double angle = std::atan2(ny, nx);
650
651 // For the left face, we want to consider angles near π and -π as equivalent.
652 double diff_left = fabs(angle - M_PI);
653 if (angle < 0)
654 diff_left = std::min(diff_left, fabs(angle + M_PI));
655
656 double diff_bottom = fabs(angle - (-M_PI / 2));
657 double diff_right = fabs(angle - 0);
658 double diff_top = fabs(angle - (M_PI / 2));
659
660 // Start by assuming left has the smallest difference.
661 int priority = 1;
662 double minDiff = diff_left;
663 if (diff_bottom < minDiff)
664 {
665 minDiff = diff_bottom;
666 priority = 2;
667 }
668 if (diff_right < minDiff)
669 {
670 minDiff = diff_right;
671 priority = 3;
672 }
673 if (diff_top < minDiff)
674 { /* minDiff = diff_top; */
675 priority = 4;
676 }
677 return priority;
678}

680// Sort_Neighbours reorders each cell's Neighbours vector so that the
681// indices (or ghost marker -1) are arranged in the following order:
682// left face, bottom face, right face, top face.
683// It uses the cell's Face_Normals (assumed to be a vector of 2D vectors)
684// to decide each face's orientation.
685void Sort_Neighbours(vector<Cell> &Cells)
686{
687 for (size_t i = 0; i < Cells.size(); i++)
688 {
689 Cell &cell = Cells[i];
690 // Assume that the number of faces is stored in cell.numFaces
691 // (or, equivalently, cell.nodeIndices.size() for many cell types).
692 int nFaces = cell.numFaces;
693
694 // Build a vector of (faceIndex, priority) pairs.
695 vector<pair<int, int>> facePriority;
696 for (int f = 0; f < nFaces; f++)
697 {
698 // Get the face normal. We assume that Face_Normals[f]
699 // is a vector<double> of at least 2 values: [nx, ny].
700 double nx = cell.Face_Normals[f * 2 + 0];
701 double ny = cell.Face_Normals[f * 2 + 1];
702 int priority = compute_face_priority(nx, ny);
703 facePriority.push_back(make_pair(f, priority));
704 }
705
706 // Sort faces based on the computed priority.
707 std::sort(facePriority.begin(), facePriority.end(),
708 [](const pair<int, int> &a, const pair<int, int> &b)
709 {
710 return a.second < b.second;
711 });
712
713 // Rebuild a sorted neighbour vector. The ghost cell value (-1)
714 // stays attached to its face.
715 vector<int> sortedNeighbours(nFaces, -1);
716 for (size_t idx = 0; idx < facePriority.size(); idx++)
717 {
718 int origFaceIndex = facePriority[idx].first;
719 sortedNeighbours[idx] = cell.Neighbours[origFaceIndex];
720 }
721
722 // Update the cell's neighbour vector.
723 cell.Neighbours = sortedNeighbours;
724 }
725}
