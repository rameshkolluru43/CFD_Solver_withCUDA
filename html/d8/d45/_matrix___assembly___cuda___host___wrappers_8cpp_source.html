62size_t estimate_matrix_assembly_memory_usage(int No_Physical_Cells, bool use_sparse)
63{
64 size_t total_memory = 0;
65
66 // Input arrays
67 total_memory += No_Physical_Cells * sizeof(double); // cell_areas
68 total_memory += No_Physical_Cells * 4 * sizeof(double); // face_areas
69 total_memory += No_Physical_Cells * 4 * sizeof(int); // neighbors
70 total_memory += No_Physical_Cells * 4 * sizeof(double); // conservative_vars
71
72 if (use_sparse)
73 {
74 // Sparse matrix format (COO)
75 int avg_neighbors = 3; // Average neighbors per cell in 2D
76 int total_nnz = No_Physical_Cells * 16 * (1 + avg_neighbors); // 4x4 entries per cell and neighbors
77 total_memory += total_nnz * sizeof(int); // row_indices
78 total_memory += total_nnz * sizeof(int); // col_indices
79 total_memory += total_nnz * sizeof(double); // values
80 total_memory += No_Physical_Cells * sizeof(int); // nnz_per_cell
81 total_memory += No_Physical_Cells * sizeof(int); // nnz_offsets
82 }
83 else
84 {
85 // Dense matrix format
86 size_t matrix_size = 4 * No_Physical_Cells;
87 total_memory += matrix_size * matrix_size * sizeof(double);
88 }
89
90 // Vector b
91 total_memory += No_Physical_Cells * 4 * sizeof(double);
92
93 return total_memory;
94}

99std::pair<int, int> get_optimal_launch_params(int No_Physical_Cells, int kernel_type)
100{
101 int device;
102 CUDA_CHECK(cudaGetDevice(&device));
103
104 cudaDeviceProp prop;
105 CUDA_CHECK(cudaGetDeviceProperties(&prop, device));
106
107 int block_size;
108 int grid_size;
109
110 switch (kernel_type)
111 {
112 case 0: // Dense matrix assembly
113 block_size = std::min(256, prop.maxThreadsPerBlock);
114 grid_size = (No_Physical_Cells + block_size - 1) / block_size;
115 break;
116
117 case 1: // Sparse matrix assembly
118 block_size = std::min(128, prop.maxThreadsPerBlock);
119 grid_size = (No_Physical_Cells + block_size - 1) / block_size;
120 break;
121
122 case 2: // Coalesced sparse matrix assembly
123 block_size = std::min(256, prop.maxThreadsPerBlock);
124 // Ensure block_size is multiple of warp size
125 block_size = (block_size / prop.warpSize) * prop.warpSize;
126 grid_size = (No_Physical_Cells * prop.warpSize + block_size - 1) / block_size;
127 break;
128
129 default:
130 block_size = 256;
131 grid_size = (No_Physical_Cells + block_size - 1) / block_size;
132 }
133
134 // Limit grid size to device maximum
135 grid_size = std::min(grid_size, prop.maxGridSize[0]);
136
137 return std::make_pair(grid_size, block_size);
138}

147double assemble_dense_matrix_cuda(
148 const std::vector<double> &cell_areas,
149 const std::vector<double> &face_areas,
150 const std::vector<int> &neighbors,
151 const std::vector<double> &conservative_vars,
152 std::vector<double> &matrix,
153 double dt,
154 int No_Physical_Cells)
155{
156
157 auto start_time = std::chrono::high_resolution_clock::now();
158
159 // Calculate matrix size
160 int matrix_size = 4 * No_Physical_Cells;
161 matrix.resize(matrix_size * matrix_size, 0.0);
162
163 // Allocate device memory
164 double *d_cell_areas, *d_face_areas, *d_conservative_vars, *d_matrix;
165 int *d_neighbors;
166
167 CUDA_CHECK(cudaMalloc(&d_cell_areas, No_Physical_Cells * sizeof(double)));
168 CUDA_CHECK(cudaMalloc(&d_face_areas, No_Physical_Cells * 4 * sizeof(double)));
169 CUDA_CHECK(cudaMalloc(&d_neighbors, No_Physical_Cells * 4 * sizeof(int)));
170 CUDA_CHECK(cudaMalloc(&d_conservative_vars, No_Physical_Cells * 4 * sizeof(double)));
171 CUDA_CHECK(cudaMalloc(&d_matrix, matrix_size * matrix_size * sizeof(double)));
172
173 // Copy data to device
174 CUDA_CHECK(cudaMemcpy(d_cell_areas, cell_areas.data(), No_Physical_Cells * sizeof(double), cudaMemcpyHostToDevice));
175 CUDA_CHECK(cudaMemcpy(d_face_areas, face_areas.data(), No_Physical_Cells * 4 * sizeof(double), cudaMemcpyHostToDevice));
176 CUDA_CHECK(cudaMemcpy(d_neighbors, neighbors.data(), No_Physical_Cells * 4 * sizeof(int), cudaMemcpyHostToDevice));
177 CUDA_CHECK(cudaMemcpy(d_conservative_vars, conservative_vars.data(), No_Physical_Cells * 4 * sizeof(double), cudaMemcpyHostToDevice));
178
179 // Initialize matrix to zero
180 auto launch_params = get_optimal_launch_params(matrix_size * matrix_size, 0);
181 initialize_matrix_kernel<<<launch_params.first, launch_params.second>>>(d_matrix, matrix_size * matrix_size);
182 CUDA_CHECK(cudaGetLastError());
183
184 // Launch matrix assembly kernel
185 launch_params = get_optimal_launch_params(No_Physical_Cells, 0);
186
187 auto kernel_start = std::chrono::high_resolution_clock::now();
188
189 assemble_dense_matrix_kernel<<<launch_params.first, launch_params.second>>>(
190 d_cell_areas, d_face_areas, d_neighbors, d_conservative_vars,
191 d_matrix, dt, No_Physical_Cells);
192
193 CUDA_CHECK(cudaDeviceSynchronize());
194 CUDA_CHECK(cudaGetLastError());
195
196 auto kernel_end = std::chrono::high_resolution_clock::now();
197
198 // Copy result back to host
199 CUDA_CHECK(cudaMemcpy(matrix.data(), d_matrix, matrix_size * matrix_size * sizeof(double), cudaMemcpyDeviceToHost));
200
201 // Cleanup device memory
202 CUDA_CHECK(cudaFree(d_cell_areas));
203 CUDA_CHECK(cudaFree(d_face_areas));
204 CUDA_CHECK(cudaFree(d_neighbors));
205 CUDA_CHECK(cudaFree(d_conservative_vars));
206 CUDA_CHECK(cudaFree(d_matrix));
207
208 auto end_time = std::chrono::high_resolution_clock::now();
209
210 double total_time = std::chrono::duration<double, std::milli>(end_time - start_time).count();
211 double kernel_time = std::chrono::duration<double, std::milli>(kernel_end - kernel_start).count();
212
213 std::cout << "Dense Matrix Assembly CUDA Performance:" << std::endl;
214 std::cout << " Total Time: " << total_time << " ms" << std::endl;
215 std::cout << " Kernel Time: " << kernel_time << " ms" << std::endl;
216 std::cout << " Memory Transfer Time: " << (total_time - kernel_time) << " ms" << std::endl;
217 std::cout << " Matrix Size: " << matrix_size << " x " << matrix_size << std::endl;
218 std::cout << " Grid/Block Size: " << launch_params.first << "/" << launch_params.second << std::endl;
219
220 return total_time;
221}

226double assemble_sparse_matrix_cuda(
227 const std::vector<double> &cell_areas,
228 const std::vector<double> &face_areas,
229 const std::vector<int> &neighbors,
230 const std::vector<double> &conservative_vars,
231 std::vector<int> &row_indices,
232 std::vector<int> &col_indices,
233 std::vector<double> &values,
234 double dt,
235 int No_Physical_Cells)
236{
237
238 auto start_time = std::chrono::high_resolution_clock::now();
239
240 // Allocate device memory for inputs
241 double *d_cell_areas, *d_face_areas, *d_conservative_vars;
242 int *d_neighbors;
243
244 CUDA_CHECK(cudaMalloc(&d_cell_areas, No_Physical_Cells * sizeof(double)));
245 CUDA_CHECK(cudaMalloc(&d_face_areas, No_Physical_Cells * 4 * sizeof(double)));
246 CUDA_CHECK(cudaMalloc(&d_neighbors, No_Physical_Cells * 4 * sizeof(int)));
247 CUDA_CHECK(cudaMalloc(&d_conservative_vars, No_Physical_Cells * 4 * sizeof(double)));

251 CUDA_CHECK(cudaMemcpy(d_face_areas, face_areas.data(), No_Physical_Cells * 4 * sizeof(double), cudaMemcpyHostToDevice));
252 CUDA_CHECK(cudaMemcpy(d_neighbors, neighbors.data(), No_Physical_Cells * 4 * sizeof(int), cudaMemcpyHostToDevice));
253 CUDA_CHECK(cudaMemcpy(d_conservative_vars, conservative_vars.data(), No_Physical_Cells * 4 * sizeof(double), cudaMemcpyHostToDevice));
254
255 // Step 1: Count non-zeros per cell
256 int *d_nnz_per_cell;
257 CUDA_CHECK(cudaMalloc(&d_nnz_per_cell, No_Physical_Cells * sizeof(int)));
258
259 auto launch_params = get_optimal_launch_params(No_Physical_Cells, 1);
260 count_nonzeros_per_cell_kernel<<<launch_params.first, launch_params.second>>>(

262 CUDA_CHECK(cudaGetLastError());
263
264 // Step 2: Compute cumulative sum to get offsets
265 thrust::device_vector<int> d_nnz_offsets(No_Physical_Cells + 1);
266 d_nnz_offsets[0] = 0;
267
268 thrust::device_ptr<int> nnz_ptr(d_nnz_per_cell);
269 thrust::inclusive_scan(nnz_ptr, nnz_ptr + No_Physical_Cells, d_nnz_offsets.begin() + 1);
270
271 // Get total number of non-zeros
272 int total_nnz;
273 CUDA_CHECK(cudaMemcpy(&total_nnz, thrust::raw_pointer_cast(d_nnz_offsets.data()) + No_Physical_Cells, sizeof(int), cudaMemcpyDeviceToHost));
274
275 // Allocate output arrays
276 row_indices.resize(total_nnz);
277 col_indices.resize(total_nnz);
278 values.resize(total_nnz);
279
280 int *d_row_indices, *d_col_indices;
281 double *d_values;
282
283 CUDA_CHECK(cudaMalloc(&d_row_indices, total_nnz * sizeof(int)));
284 CUDA_CHECK(cudaMalloc(&d_col_indices, total_nnz * sizeof(int)));
285 CUDA_CHECK(cudaMalloc(&d_values, total_nnz * sizeof(double)));
286
287 // Step 3: Assemble sparse matrix
288 auto kernel_start = std::chrono::high_resolution_clock::now();
289
290 assemble_sparse_matrix_kernel<<<launch_params.first, launch_params.second>>>(
291 d_cell_areas, d_face_areas, d_neighbors, d_conservative_vars,
292 thrust::raw_pointer_cast(d_nnz_offsets.data()),
293 d_row_indices, d_col_indices, d_values,
294 dt, No_Physical_Cells);
295
296 CUDA_CHECK(cudaDeviceSynchronize());
297 CUDA_CHECK(cudaGetLastError());
298
299 auto kernel_end = std::chrono::high_resolution_clock::now();
300
301 // Copy results back to host
302 CUDA_CHECK(cudaMemcpy(row_indices.data(), d_row_indices, total_nnz * sizeof(int), cudaMemcpyDeviceToHost));
303 CUDA_CHECK(cudaMemcpy(col_indices.data(), d_col_indices, total_nnz * sizeof(int), cudaMemcpyDeviceToHost));
304 CUDA_CHECK(cudaMemcpy(values.data(), d_values, total_nnz * sizeof(double), cudaMemcpyDeviceToHost));
305
306 // Cleanup device memory
307 CUDA_CHECK(cudaFree(d_cell_areas));
308 CUDA_CHECK(cudaFree(d_face_areas));
309 CUDA_CHECK(cudaFree(d_neighbors));
310 CUDA_CHECK(cudaFree(d_conservative_vars));
311 CUDA_CHECK(cudaFree(d_nnz_per_cell));
312 CUDA_CHECK(cudaFree(d_row_indices));
313 CUDA_CHECK(cudaFree(d_col_indices));
314 CUDA_CHECK(cudaFree(d_values));
315
316 auto end_time = std::chrono::high_resolution_clock::now();
317
318 double total_time = std::chrono::duration<double, std::milli>(end_time - start_time).count();
319 double kernel_time = std::chrono::duration<double, std::milli>(kernel_end - kernel_start).count();
320
321 std::cout << "Sparse Matrix Assembly CUDA Performance:" << std::endl;
322 std::cout << " Total Time: " << total_time << " ms" << std::endl;
323 std::cout << " Kernel Time: " << kernel_time << " ms" << std::endl;
324 std::cout << " Memory Transfer Time: " << (total_time - kernel_time) << " ms" << std::endl;
325 std::cout << " Total Non-zeros: " << total_nnz << std::endl;
326 std::cout << " Sparsity: " << (1.0 - (double)total_nnz / ((double)4 * No_Physical_Cells * 4 * No_Physical_Cells)) * 100 << "%" << std::endl;
327 std::cout << " Grid/Block Size: " << launch_params.first << "/" << launch_params.second << std::endl;
328
329 return total_time;
