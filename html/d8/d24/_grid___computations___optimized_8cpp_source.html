51// Fast inverse square root approximation (Quake III algorithm)
52inline float fast_inv_sqrt(float x)
53{
54 float xhalf = 0.5f * x;
55 int i = *(int *)&x;
56 i = 0x5f3759df - (i >> 1);
57 x = *(float *)&i;
58 x = x * (1.5f - xhalf * x * x);
59 return x;
60}

63inline double distance_squared(const V_D &P1, const V_D &P2)
64{
65 double dx = P2[0] - P1[0];
66 double dy = P2[1] - P1[1];
67 double dz = P2[2] - P1[2];
68 return dx * dx + dy * dy + dz * dz;

72#ifdef __AVX2__
73inline void simd_cross_product(const double *a, const double *b, double *result)
74{
75 // Load vectors into AVX registers
76 __m256d va = _mm256_loadu_pd(a);
77 __m256d vb = _mm256_loadu_pd(b);
78
79 // Perform cross product using AVX operations
80 // result[0] = a[1]*b[2] - a[2]*b[1]
81 // result[1] = a[2]*b[0] - a[0]*b[2]
82 // result[2] = a[0]*b[1] - a[1]*b[0]
83
84 __m256d temp1 = _mm256_permute4x64_pd(va, 0xC9); // [a1, a2, a0, a3]
85 __m256d temp2 = _mm256_permute4x64_pd(vb, 0xD2); // [b2, b0, b1, b3]
86 __m256d temp3 = _mm256_permute4x64_pd(va, 0xD2); // [a2, a0, a1, a3]
87 __m256d temp4 = _mm256_permute4x64_pd(vb, 0xC9); // [b1, b2, b0, b3]
88
89 __m256d prod1 = _mm256_mul_pd(temp1, temp2);
90 __m256d prod2 = _mm256_mul_pd(temp3, temp4);
91 __m256d cross = _mm256_sub_pd(prod1, prod2);
92
93 _mm256_storeu_pd(result, cross);
94}

116// Optimized cell construction with move semantics
117void Construct_Cell_Optimized(Cell &&Grid_Cell)
118{
119 // Use references to avoid copies
120 auto &vertices = Grid_Cell.Cell_Vertices;
121
122 // Pre-allocate vectors
123 Grid_Cell.Cell_Center.resize(3);
124 Grid_Cell.Face_Areas.reserve(4);
125 Grid_Cell.Face_Normals.reserve(8);

127 // Compute centroid efficiently
128 double cx = 0.0, cy = 0.0, cz = 0.0;
129 for (size_t i = 0; i < vertices.size(); i += 3)
130 {
131 cx += vertices[i];
132 cy += vertices[i + 1];
133 cz += vertices[i + 2];
134 }

137 const double inv_vertices = 1.0 / num_vertices;
138 Grid_Cell.Cell_Center[0] = cx * inv_vertices;
139 Grid_Cell.Cell_Center[1] = cy * inv_vertices;
140 Grid_Cell.Cell_Center[2] = cz * inv_vertices;
141
142 // Construct faces with optimized calculations
143 Construct_Face_Optimized(Grid_Cell);
144
145 // Calculate area using optimized cross product
146 if (vertices.size() == 12)
147 { // Quadrilateral
148 double diag1_x = vertices[6] - vertices[0];
149 double diag1_y = vertices[7] - vertices[1];
150 double diag2_x = vertices[9] - vertices[3];
151 double diag2_y = vertices[10] - vertices[4];

154 Grid_Cell.Area = 0.5 * std::abs(diag1_x * diag2_y - diag1_y * diag2_x);

161void Construct_Face_Optimized(Cell &Grid_Cell)
162{
163 const auto &vertices = Grid_Cell.Cell_Vertices;
164
165 // Extract points efficiently
166 const double *p0 = &vertices[0]; // [x0, y0, z0]
167 const double *p1 = &vertices[3]; // [x1, y1, z1]
168 const double *p2 = &vertices[6]; // [x2, y2, z2]
169 const double *p3 = &vertices[9]; // [x3, y3, z3]

171 // Construct faces with SIMD operations where available

174 double dx = b[0] - a[0];

178 Grid_Cell.Face_Areas.push_back(length);

184 construct_face_segment(p3, p0);
185 construct_face_segment(p0, p1);
186 construct_face_segment(p1, p2);
187 construct_face_segment(p2, p3);

258// Fast angle calculation using atan2 approximation
259inline double fast_atan2(double y, double x)
260{
261 // Fast atan2 approximation using polynomial approximation
262 const double abs_y = std::abs(y);
263 const double abs_x = std::abs(x);
264
265 if (abs_x < 1e-10 && abs_y < 1e-10)
266 return 0.0;
267
268 double a = std::min(abs_x, abs_y) / std::max(abs_x, abs_y);
269 double s = a * a;
270 double r = ((-0.0464964749 * s + 0.15931422) * s - 0.327622764) * s * a + a;
271
272 if (abs_y > abs_x)
273 r = 1.57079637 - r;
274 if (x < 0)
275 r = 3.14159274 - r;
276 if (y < 0)
277 r = -r;
278
279 return r;
280}

282// Optimized anti-clockwise sorting using faster angle calculation
283void Sort_Points_AntiClockWise_Optimized(V_D &points, V_I &indices)
284{
285 if (points.empty() || points.size() % 3 != 0)
286 return;
287
288 GridPerformance::start_timer();
289
290 // Compute centroid
291 V_D centroid(3, 0.0);
292 Compute_Centroid_Optimized(points, centroid);
293
294 // Create point-index pairs
295 std::vector<std::pair<std::array<double, 3>, int>> point_pairs;
296 point_pairs.reserve(points.size() / 3);
297
298 for (size_t i = 0; i < points.size(); i += 3)
299 {
300 std::array<double, 3> point = {points[i], points[i + 1], points[i + 2]};
301 point_pairs.emplace_back(std::move(point), indices[i / 3]);
302 }
303
304 // Sort using optimized angle calculation
305 std::sort(point_pairs.begin(), point_pairs.end(),
306 [&centroid](const auto &a, const auto &b)
307 {
308 double angle_a = fast_atan2(a.first[1] - centroid[1], a.first[0] - centroid[0]);
309 double angle_b = fast_atan2(b.first[1] - centroid[1], b.first[0] - centroid[0]);
310 return angle_a < angle_b;
311 });
312
313 // Rebuild vectors
314 points.clear();
315 indices.clear();
316 points.reserve(point_pairs.size() * 3);
317 indices.reserve(point_pairs.size());
318
319 for (const auto &pair : point_pairs)
320 {
321 points.insert(points.end(), pair.first.begin(), pair.first.end());
322 indices.push_back(pair.second);
323 }
324
325 GridPerformance::end_timer("Optimized Anti-clockwise Sorting");
326}
