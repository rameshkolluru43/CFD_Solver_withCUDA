1063// To check if the points are sorted in anti-clockwise order
1064bool Are_Points_Sorted_AntiClockWise(const std::vector<V_D> &groupedPoints, const V_D &Centroid)
1065{
1066 for (size_t i = 1; i < groupedPoints.size(); ++i)
1067 {
1068 double anglePrev = atan2(groupedPoints[i - 1][1] - Centroid[1], groupedPoints[i - 1][0] - Centroid[0]);
1069 double angleCurr = atan2(groupedPoints[i][1] - Centroid[1], groupedPoints[i][0] - Centroid[0]);
1070 if (anglePrev > angleCurr) // If any point is out of order
1071 {
1072 return false;
1073 }
1074 }
1075 return true;

1078void Sort_Points_AntiClockWise(V_D &Points)
1079{
1080 // Check if the input is valid
1081 if (Points.empty() || Points.size() % 3 != 0)
1082 {
1083 std::cerr << "Error: Points vector must be non-empty and have a size that is a multiple of 3." << std::endl;
1084 return;
1085 }
1086
1087 V_D Centroid(3, 0.0);
1088 Compute_Centroid(Points, Centroid);
1089
1090 // Grouping Points into 3D points
1091 std::vector<V_D> groupedPoints;
1092 for (size_t i = 0; i < Points.size(); i += 3)
1093 {
1094 groupedPoints.push_back({Points[i], Points[i + 1], Points[i + 2]});
1095 }
1096
1097 // Check if points are already sorted
1098 if (!Are_Points_Sorted_AntiClockWise(groupedPoints, Centroid))
1099 {
1100 // Sorting based on angles from the centroid in the XY plane
