421 void compute_flow_reynolds_numbers(const FluidProperties &props,
422 const VelocityField &vel_field,
423 double characteristic_length,
424 double &Re_global, double &Re_cell)
425 {
426
427 // Compute characteristic velocity
428 double u_max = 0.0, u_avg = 0.0;
429 int valid_cells = 0;
430
431 for (size_t i = 0; i < vel_field.u.size(); ++i)
432 {
433 if (i < vel_field.v.size())
434 {
435 double u_mag = std::sqrt(vel_field.u[i] * vel_field.u[i] + vel_field.v[i] * vel_field.v[i]);
436 if (!vel_field.w.empty() && i < vel_field.w.size())
437 {
438 u_mag = std::sqrt(u_mag * u_mag + vel_field.w[i] * vel_field.w[i]);
439 }
440
441 u_max = std::max(u_max, u_mag);
442 u_avg += u_mag;
443 valid_cells++;
444 }
445 }
446
447 if (valid_cells > 0)
448 {
449 u_avg /= valid_cells;
450 }
451
452 // Global Reynolds number based on characteristic length and average velocity
453 Re_global = (props.kinematic_viscosity > 0.0) ? (u_avg * characteristic_length) / props.kinematic_viscosity : 0.0;
454
455 // Cell Reynolds number based on typical cell size
456 double typical_cell_size = 0.0;
457 int cells_counted = 0;
458
459 for (int i = 0; i < std::min(No_Physical_Cells, static_cast<int>(Cells.size())); ++i)
460 {
461 double cell_size = (Cells[i].Vol > 0.0) ? std::pow(Cells[i].Vol, 1.0 / 3.0) : std::sqrt(Cells[i].Area);
462 if (cell_size > 0.0)
463 {
464 typical_cell_size += cell_size;
465 cells_counted++;
466 }
467 }
468
469 if (cells_counted > 0)
470 {
471 typical_cell_size /= cells_counted;
472 }
473
474 Re_cell = (props.kinematic_viscosity > 0.0 && typical_cell_size > 0.0) ? (u_max * typical_cell_size) / props.kinematic_viscosity : 0.0;
475
476 std::cout << "Reynolds number analysis:" << std::endl;
477 std::cout << " - Characteristic length: " << characteristic_length << " m" << std::endl;
478 std::cout << " - Typical cell size: " << typical_cell_size << " m" << std::endl;
479 std::cout << " - Average velocity: " << u_avg << " m/s" << std::endl;
480 std::cout << " - Maximum velocity: " << u_max << " m/s" << std::endl;
481 std::cout << " - Global Reynolds number: " << Re_global << std::endl;
482 std::cout << " - Cell Reynolds number: " << Re_cell << std::endl;
483
484 // Provide guidance based on Reynolds numbers
485 if (Re_global < 1.0)
486 {
487 std::cout << " - Flow regime: Highly viscous (Stokes flow)" << std::endl;
488 }
489 else if (Re_global < 100.0)
490 {
491 std::cout << " - Flow regime: Laminar, viscous-dominated" << std::endl;
492 }
493 else if (Re_global < 2000.0)
494 {
495 std::cout << " - Flow regime: Laminar" << std::endl;
496 }
497 else if (Re_global < 4000.0)
498 {
499 std::cout << " - Flow regime: Transitional" << std::endl;
500 }
501 else
502 {
503 std::cout << " - Flow regime: Turbulent (consider turbulence modeling)" << std::endl;
504 }
505
506 if (Re_cell > 2.0)
507 {
508 std::cout << " - Warning: Cell Reynolds number > 2. Consider refining the grid or using upwind schemes." << std::endl;
509 }
510 }
