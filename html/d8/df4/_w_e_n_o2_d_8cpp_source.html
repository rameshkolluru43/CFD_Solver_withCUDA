9void WENO_Reconstruction(double &a, double &b, double &c, double &d, double &e, int &shift, double &U)
10{
11 // Check for NaN or infinite values in input
12 if (!std::isfinite(a) || !std::isfinite(b) || !std::isfinite(c) || !std::isfinite(d) || !std::isfinite(e))
13 {
14 std::cout << "Warning: Non-finite values in WENO reconstruction input" << std::endl;
15 U = c; // Fall back to central value
16 return;
17 }
18
19 double d0 = 0.0, d1 = 0.0, d2 = 0.0, b0 = 0.0, b1 = 0.0, b2 = 0.0, a0 = 0.0, a1 = 0.0, a2 = 0.0;
20 double v0 = 0.0, v1 = 0.0, v2 = 0.0, w0 = 0.0, w1 = 0.0, w2 = 0.0, sum = 0.0, epsilon = 1e-6;
21
22 int p = 2;
23 switch (shift)
24 {
25 case 0: // for Left Values
26 d0 = 3.0 / 10.0;
27 d1 = 6.0 / 10.0;
28 d2 = 1.0 / 10.0;
29
30 v2 = (2.0 * a - 7.0 * b + 11.0 * c) / 6.0;
31 v1 = (-1.0 * b + 5.0 * c + 2.0 * d) / 6.0;
32 v0 = (2.0 * c + 5.0 * d - 1.0 * e) / 6.0;
33
34 break;
35 case 1: // for Right Values
36 d0 = 1.0 / 10.0;
37 d1 = 6.0 / 10.0;
38 d2 = 3.0 / 10.0;
39
40 v2 = (-1.0 * a + 5.0 * b + 2.0 * c) / 6.0;
41 v1 = (2.0 * b + 5.0 * c - 1.0 * d) / 6.0;
42 v0 = (11.0 * c - 7.0 * d + 2.0 * e) / 6.0;
43
44 break;
45 }
46
47 b2 = (13.0 / 12.0) * pow((a - 2.0 * b + c), 2) + (1.0 / 4.0) * pow((a - 4.0 * b + 3.0 * c), 2);
48
49 b1 = (13.0 / 12.0) * pow((b - 2.0 * c + d), 2) + (1.0 / 4.0) * pow((b - d), 2);
50
51 b0 = (13.0 / 12.0) * pow((c - 2.0 * d + e), 2) + (1.0 / 4.0) * pow((3.0 * c - 4.0 * d + e), 2);
52
53 a0 = d0 / pow((epsilon + b0), p);
54 a1 = d1 / pow((epsilon + b1), p);
55 a2 = d2 / pow((epsilon + b2), p);
56
57 sum = a0 + a1 + a2;
58
59 // Check for degenerate case where sum is too small
60 if (sum < epsilon)
61 {
62 // Fall back to simple average of the three candidate values
63 U = (v0 + v1 + v2) / 3.0;
64 }
65 else
66 {
67 w0 = a0 / sum;
68 w1 = a1 / sum;
69 w2 = a2 / sum;
70 U = w0 * v0 + w1 * v1 + w2 * v2;
71 }
72
73 // Final validation
74 if (!std::isfinite(U))
75 {
76 std::cout << "Warning: Non-finite result in WENO reconstruction, using central value" << std::endl;
77 U = c;
78 }
