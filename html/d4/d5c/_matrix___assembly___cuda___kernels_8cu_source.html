89__device__ void compute_flux_jacobian_device(
90 const double* cell_state, // Conservative variables [rho, rho*u, rho*v, E]
91 double* jacobian, // Output 4x4 Jacobian matrix (flattened)
92 int face_direction, // 0=left, 1=bottom, 2=right, 3=top
93 double face_area)
94{
95 // Extract conservative variables
96 double rho = cell_state[0];
97 double rho_u = cell_state[1];
98 double rho_v = cell_state[2];
99 double E = cell_state[3];
100
101 // Avoid division by zero
102 if (rho < 1e-12) rho = 1e-12;
103
104 double u = rho_u / rho;
105 double v = rho_v / rho;
106 double gamma = 1.4; // Specific heat ratio for air
107
108 // Compute pressure using ideal gas law
109 double kinetic_energy = 0.5 * (u*u + v*v);
110 double p = (gamma - 1.0) * (E - rho * kinetic_energy);
111
112 // Ensure positive pressure
113 if (p < 1e-12) p = 1e-12;
114
115 double c = sqrt(gamma * p / rho); // Speed of sound
116
117 // Zero out jacobian
118 for (int i = 0; i < JACOBIAN_SIZE; i++) {
119 jacobian[i] = 0.0;
120 }
121
122 // Compute Jacobian based on face direction
123 if (face_direction == 0 || face_direction == 2) { // x-direction faces
124 // Flux Jacobian for x-direction: ∂F_x/∂U
125 // This is a simplified Euler flux Jacobian
126 double factor = (face_direction == 2) ? face_area : -face_area;
127
128 // Row 0: ∂(ρu)/∂U
129 jacobian[0*4 + 0] = 0.0;
130 jacobian[0*4 + 1] = factor;
131 jacobian[0*4 + 2] = 0.0;
132 jacobian[0*4 + 3] = 0.0;
133
134 // Row 1: ∂(ρu² + p)/∂U
135 jacobian[1*4 + 0] = factor * (0.5*(gamma-1.0)*(u*u + v*v) - u*u);
136 jacobian[1*4 + 1] = factor * (u*(3.0-gamma));
137 jacobian[1*4 + 2] = factor * (-v*(gamma-1.0));
138 jacobian[1*4 + 3] = factor * (gamma-1.0);
139
140 // Row 2: ∂(ρuv)/∂U
141 jacobian[2*4 + 0] = factor * (-u*v);
142 jacobian[2*4 + 1] = factor * v;
143 jacobian[2*4 + 2] = factor * u;
144 jacobian[2*4 + 3] = 0.0;
145
146 // Row 3: ∂(u(E + p))/∂U
147 jacobian[3*4 + 0] = factor * u * (0.5*(gamma-1.0)*(u*u + v*v) - E/rho - p/rho);
148 jacobian[3*4 + 1] = factor * (E/rho + p/rho - 0.5*(gamma-1.0)*(u*u - v*v));
149 jacobian[3*4 + 2] = factor * (-u*v*(gamma-1.0));
150 jacobian[3*4 + 3] = factor * u * gamma;
151
152 } else { // y-direction faces
153 // Flux Jacobian for y-direction: ∂F_y/∂U
154 double factor = (face_direction == 3) ? face_area : -face_area;
155
156 // Row 0: ∂(ρv)/∂U
157 jacobian[0*4 + 0] = 0.0;
158 jacobian[0*4 + 1] = 0.0;
159 jacobian[0*4 + 2] = factor;
160 jacobian[0*4 + 3] = 0.0;
161
162 // Row 1: ∂(ρuv)/∂U
163 jacobian[1*4 + 0] = factor * (-u*v);
164 jacobian[1*4 + 1] = factor * v;
165 jacobian[1*4 + 2] = factor * u;
166 jacobian[1*4 + 3] = 0.0;
167
168 // Row 2: ∂(ρv² + p)/∂U
169 jacobian[2*4 + 0] = factor * (0.5*(gamma-1.0)*(u*u + v*v) - v*v);
170 jacobian[2*4 + 1] = factor * (-u*(gamma-1.0));
171 jacobian[2*4 + 2] = factor * (v*(3.0-gamma));
172 jacobian[2*4 + 3] = factor * (gamma-1.0);
173
174 // Row 3: ∂(v(E + p))/∂U
175 jacobian[3*4 + 0] = factor * v * (0.5*(gamma-1.0)*(u*u + v*v) - E/rho - p/rho);
176 jacobian[3*4 + 1] = factor * (-u*v*(gamma-1.0));
177 jacobian[3*4 + 2] = factor * (E/rho + p/rho - 0.5*(gamma-1.0)*(v*v - u*u));
178 jacobian[3*4 + 3] = factor * v * gamma;
179 }
180}

234__global__ void assemble_dense_matrix_kernel(
235 const double* d_cell_areas, // [No_Physical_Cells] Cell areas
236 const double* d_face_areas, // [No_Physical_Cells * 4] Face areas per cell
237 const int* d_neighbors, // [No_Physical_Cells * 4] Neighbor indices
238 const double* d_conservative_vars, // [No_Physical_Cells * 4] Conservative variables
239 double* d_matrix, // [4*No_Physical_Cells * 4*No_Physical_Cells] Output matrix
240 double dt, // Time step
241 int No_Physical_Cells) // Number of physical cells
242{
243 int cell_idx = blockIdx.x * blockDim.x + threadIdx.x;
244
245 if (cell_idx >= No_Physical_Cells) return;
246
247 // Get cell properties
248 double omega = d_cell_areas[cell_idx];
249 if (omega <= 0.0) return; // Skip invalid cells
250
251 // Get face areas for this cell
252 double face_areas[4];
253 for (int f = 0; f < 4; f++) {
254 face_areas[f] = d_face_areas[cell_idx * 4 + f];
255 if (face_areas[f] <= 0.0) face_areas[f] = 1e-12; // Avoid division by zero
256 }
257
258 // Get neighbors
259 int neighbors[4];
260 for (int f = 0; f < 4; f++) {
261 neighbors[f] = d_neighbors[cell_idx * 4 + f];
262 }
263
264 // Get conservative variables for this cell
265 double cell_state[4];
266 for (int var = 0; var < 4; var++) {
267 cell_state[var] = d_conservative_vars[cell_idx * 4 + var];
268 }
269
270 // Temporary storage for Jacobians
271 double A_x_L[JACOBIAN_SIZE], A_x_R[JACOBIAN_SIZE];
272 double A_y_B[JACOBIAN_SIZE], A_y_T[JACOBIAN_SIZE];
273
274 // Compute flux Jacobians for each face
275 compute_flux_jacobian_device(cell_state, A_x_L, 0, face_areas[0]); // Left face
276 compute_flux_jacobian_device(cell_state, A_x_R, 2, face_areas[2]); // Right face
277 compute_flux_jacobian_device(cell_state, A_y_B, 1, face_areas[1]); // Bottom face
278 compute_flux_jacobian_device(cell_state, A_y_T, 3, face_areas[3]); // Top face
279
280 int matrix_size = 4 * No_Physical_Cells;
281
282 // Add self contribution to the matrix
283 for (int row = 0; row < 4; row++) {
284 for (int col = 0; col < 4; col++) {
285 int global_row = 4 * cell_idx + row;
286 int global_col = 4 * cell_idx + col;
287
288 // Time derivative contribution (diagonal term)
289 double time_contrib = (row == col) ? omega / dt : 0.0;
290
291 // Flux Jacobian contribution
292 double flux_contrib = (dt / omega) * (
293 A_x_R[row * 4 + col] - A_x_L[row * 4 + col] +
294 A_y_T[row * 4 + col] - A_y_B[row * 4 + col]
295 );
296
297 double total_contrib = time_contrib + flux_contrib;
298
299 // Add to global matrix
300 add_matrix_contribution_atomic(d_matrix, global_row, global_col, total_contrib, matrix_size);
301 }
302 }
303
304 // Add neighbor contributions
305 double A_neighbor[JACOBIAN_SIZE];
306
307 // Right neighbor (face 2)
308 if (neighbors[2] >= 0 && neighbors[2] < No_Physical_Cells) {
309 compute_flux_jacobian_device(cell_state, A_neighbor, 2, face_areas[2]);
310 for (int row = 0; row < 4; row++) {
311 for (int col = 0; col < 4; col++) {
312 int global_row = 4 * cell_idx + row;
313 int global_col = 4 * neighbors[2] + col;
314 double contrib = (dt / omega) * A_neighbor[row * 4 + col];
315 add_matrix_contribution_atomic(d_matrix, global_row, global_col, contrib, matrix_size);
316 }
317 }
318 }
319
320 // Left neighbor (face 0)
321 if (neighbors[0] >= 0 && neighbors[0] < No_Physical_Cells) {
322 compute_flux_jacobian_device(cell_state, A_neighbor, 0, face_areas[0]);
323 for (int row = 0; row < 4; row++) {
324 for (int col = 0; col < 4; col++) {
325 int global_row = 4 * cell_idx + row;
326 int global_col = 4 * neighbors[0] + col;
327 double contrib = -(dt / omega) * A_neighbor[row * 4 + col];
328 add_matrix_contribution_atomic(d_matrix, global_row, global_col, contrib, matrix_size);
329 }
330 }
331 }
332
333 // Top neighbor (face 3)
334 if (neighbors[3] >= 0 && neighbors[3] < No_Physical_Cells) {
335 compute_flux_jacobian_device(cell_state, A_neighbor, 3, face_areas[3]);
336 for (int row = 0; row < 4; row++) {
337 for (int col = 0; col < 4; col++) {
338 int global_row = 4 * cell_idx + row;
339 int global_col = 4 * neighbors[3] + col;
340 double contrib = (dt / omega) * A_neighbor[row * 4 + col];
341 add_matrix_contribution_atomic(d_matrix, global_row, global_col, contrib, matrix_size);
342 }
343 }
344 }
345
346 // Bottom neighbor (face 1)
347 if (neighbors[1] >= 0 && neighbors[1] < No_Physical_Cells) {
348 compute_flux_jacobian_device(cell_state, A_neighbor, 1, face_areas[1]);
349 for (int row = 0; row < 4; row++) {
350 for (int col = 0; col < 4; col++) {
351 int global_row = 4 * cell_idx + row;
352 int global_col = 4 * neighbors[1] + col;
353 double contrib = -(dt / omega) * A_neighbor[row * 4 + col];
354 add_matrix_contribution_atomic(d_matrix, global_row, global_col, contrib, matrix_size);
355 }
356 }
357 }

549__global__ void assemble_sparse_matrix_coalesced_kernel(
550 const double* d_cell_areas,
551 const double* d_face_areas,
552 const int* d_neighbors,
553 const double* d_conservative_vars,
554 const int* d_nnz_offsets,
555 int* d_row_indices,
556 int* d_col_indices,
557 double* d_values,
558 double dt,
559 int No_Physical_Cells)
560{
561 int warp_id = (blockIdx.x * blockDim.x + threadIdx.x) / 32;
562 int lane_id = threadIdx.x % 32;
563 int total_warps = (gridDim.x * blockDim.x + 31) / 32;
564
565 // Each warp processes multiple cells
566 for (int cell_idx = warp_id; cell_idx < No_Physical_Cells; cell_idx += total_warps) {
567
568 if (cell_idx >= No_Physical_Cells) return;
569
570 // Shared memory for Jacobians (per warp)
571 __shared__ double shared_jacobians[32 * JACOBIAN_SIZE];
572 double* my_jacobian = &shared_jacobians[lane_id * JACOBIAN_SIZE];
573
574 // Load cell data cooperatively
575 double omega = d_cell_areas[cell_idx];
576 if (omega <= 0.0) continue;
577
578 // Each thread in warp loads different data elements
579 double face_areas[4];
580 int neighbors[4];
581 double cell_state[4];
582
583 // Coalesced loading
584 if (lane_id < 4) {
585 face_areas[lane_id] = d_face_areas[cell_idx * 4 + lane_id];
586 neighbors[lane_id] = d_neighbors[cell_idx * 4 + lane_id];
587 cell_state[lane_id] = d_conservative_vars[cell_idx * 4 + lane_id];
588 }
589
590 // Broadcast within warp
591 for (int i = 0; i < 4; i++) {
592 face_areas[i] = __shfl_sync(0xFFFFFFFF, face_areas[i], i);
593 neighbors[i] = __shfl_sync(0xFFFFFFFF, neighbors[i], i);
594 cell_state[i] = __shfl_sync(0xFFFFFFFF, cell_state[i], i);
595 }
596
597 // Compute Jacobians in parallel
598 if (lane_id == 0) compute_flux_jacobian_device(cell_state, my_jacobian, 0, face_areas[0]);
599 else if (lane_id == 1) compute_flux_jacobian_device(cell_state, my_jacobian, 1, face_areas[1]);
600 else if (lane_id == 2) compute_flux_jacobian_device(cell_state, my_jacobian, 2, face_areas[2]);
601 else if (lane_id == 3) compute_flux_jacobian_device(cell_state, my_jacobian, 3, face_areas[3]);
602
603 __syncwarp();
604
605 // Assembly phase - only one thread per warp does this to avoid race conditions
606 if (lane_id == 0) {
607 int entry_offset = d_nnz_offsets[cell_idx];
608 int entry_idx = entry_offset;
609
610 // Self contributions
611 for (int row = 0; row < 4; row++) {
612 for (int col = 0; col < 4; col++) {
613 d_row_indices[entry_idx] = 4 * cell_idx + row;
614 d_col_indices[entry_idx] = 4 * cell_idx + col;
615
616 double time_contrib = (row == col) ? omega / dt : 0.0;
617 double flux_contrib = (dt / omega) * (
618 shared_jacobians[2 * JACOBIAN_SIZE + row * 4 + col] - shared_jacobians[0 * JACOBIAN_SIZE + row * 4 + col] +
619 shared_jacobians[3 * JACOBIAN_SIZE + row * 4 + col] - shared_jacobians[1 * JACOBIAN_SIZE + row * 4 + col]
620 );
621
622 d_values[entry_idx] = time_contrib + flux_contrib;
623 entry_idx++;
624 }
625 }
626
627 // Neighbor contributions
628 int face_signs[4] = {-1, -1, 1, 1};
629 for (int face = 0; face < 4; face++) {
630 if (neighbors[face] >= 0 && neighbors[face] < No_Physical_Cells) {
631 for (int row = 0; row < 4; row++) {
632 for (int col = 0; col < 4; col++) {
633 d_row_indices[entry_idx] = 4 * cell_idx + row;
634 d_col_indices[entry_idx] = 4 * neighbors[face] + col;
635 d_values[entry_idx] = face_signs[face] * (dt / omega) * shared_jacobians[face * JACOBIAN_SIZE + row * 4 + col];
636 entry_idx++;
637 }
638 }
639 }
640 }
641 }
642 }
643}
