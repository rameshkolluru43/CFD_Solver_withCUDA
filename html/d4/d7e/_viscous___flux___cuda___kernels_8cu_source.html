27__global__ void calculate_viscous_stress_kernel(
28 const double* velocity_gradients, // [num_cells * 9] - Velocity gradient tensor
29 const double* temperature_gradients, // [num_cells * 3] - Temperature gradients
30 const double* primitive_vars, // [num_cells * 5] - Primitive variables (rho,u,v,w,P)
31 const double* face_normals, // [num_faces * 3] - Face normal vectors
32 const double* face_areas, // [num_faces] - Face areas
33 const int* face_cells, // [num_faces * 2] - Left and right cell indices
34 double* viscous_fluxes, // [num_faces * 5] - Output viscous fluxes
35 int num_faces
36) {
37 int face_idx = blockIdx.x * blockDim.x + threadIdx.x;
38 if (face_idx >= num_faces) return;
39
40 int left_cell = face_cells[face_idx * 2 + 0];
41 int right_cell = face_cells[face_idx * 2 + 1];
42
43 // Face normal and area
44 double nx = face_normals[face_idx * 3 + 0];
45 double ny = face_normals[face_idx * 3 + 1];
46 double nz = face_normals[face_idx * 3 + 2];
47 double area = face_areas[face_idx];
48
49 // Average velocity gradients at face
50 double dudx, dudy, dudz, dvdx, dvdy, dvdz, dwdx, dwdy, dwdz;
51 double dTdx, dTdy, dTdz;
52
53 if (right_cell >= 0) {
54 // Internal face - average gradients
55 dudx = 0.5 * (velocity_gradients[left_cell * 9 + 0] + velocity_gradients[right_cell * 9 + 0]);
56 dudy = 0.5 * (velocity_gradients[left_cell * 9 + 1] + velocity_gradients[right_cell * 9 + 1]);
57 dudz = 0.5 * (velocity_gradients[left_cell * 9 + 2] + velocity_gradients[right_cell * 9 + 2]);
58 dvdx = 0.5 * (velocity_gradients[left_cell * 9 + 3] + velocity_gradients[right_cell * 9 + 3]);
59 dvdy = 0.5 * (velocity_gradients[left_cell * 9 + 4] + velocity_gradients[right_cell * 9 + 4]);
60 dvdz = 0.5 * (velocity_gradients[left_cell * 9 + 5] + velocity_gradients[right_cell * 9 + 5]);
61 dwdx = 0.5 * (velocity_gradients[left_cell * 9 + 6] + velocity_gradients[right_cell * 9 + 6]);
62 dwdy = 0.5 * (velocity_gradients[left_cell * 9 + 7] + velocity_gradients[right_cell * 9 + 7]);
63 dwdz = 0.5 * (velocity_gradients[left_cell * 9 + 8] + velocity_gradients[right_cell * 9 + 8]);
64
65 dTdx = 0.5 * (temperature_gradients[left_cell * 3 + 0] + temperature_gradients[right_cell * 3 + 0]);
66 dTdy = 0.5 * (temperature_gradients[left_cell * 3 + 1] + temperature_gradients[right_cell * 3 + 1]);
67 dTdz = 0.5 * (temperature_gradients[left_cell * 3 + 2] + temperature_gradients[right_cell * 3 + 2]);
68 } else {
69 // Boundary face - use left cell gradients
70 dudx = velocity_gradients[left_cell * 9 + 0];
71 dudy = velocity_gradients[left_cell * 9 + 1];
72 dudz = velocity_gradients[left_cell * 9 + 2];
73 dvdx = velocity_gradients[left_cell * 9 + 3];
74 dvdy = velocity_gradients[left_cell * 9 + 4];
75 dvdz = velocity_gradients[left_cell * 9 + 5];
76 dwdx = velocity_gradients[left_cell * 9 + 6];
77 dwdy = velocity_gradients[left_cell * 9 + 7];
78 dwdz = velocity_gradients[left_cell * 9 + 8];
79
80 dTdx = temperature_gradients[left_cell * 3 + 0];
81 dTdy = temperature_gradients[left_cell * 3 + 1];
82 dTdz = temperature_gradients[left_cell * 3 + 2];
83 }
84
85 // Get average temperature and calculate transport properties
86 double T_avg = 0.5 * (primitive_vars[left_cell * 5 + 4] / (primitive_vars[left_cell * 5 + 0] * R));
87 if (right_cell >= 0) {
88 T_avg += 0.5 * (primitive_vars[right_cell * 5 + 4] / (primitive_vars[right_cell * 5 + 0] * R));
89 }
90
91 double mu = calculate_viscosity(T_avg);
92 double k = calculate_thermal_conductivity(T_avg, mu);
93 double lambda = -2.0/3.0 * mu; // Bulk viscosity
94
95 // Calculate velocity divergence
96 double div_v = dudx + dvdy + dwdz;
97
98 // Calculate stress tensor components
99 double Txx = mu * (2.0 * dudx + lambda * div_v);
100 double Tyy = mu * (2.0 * dvdy + lambda * div_v);
101 double Tzz = mu * (2.0 * dwdz + lambda * div_v);
102 double Txy = mu * (dudy + dvdx);
103 double Txz = mu * (dudz + dwdx);
104 double Tyz = mu * (dvdz + dwdy);
105
106 // Calculate heat flux components
107 double qx = -k * dTdx;
108 double qy = -k * dTdy;
109 double qz = -k * dTdz;
110
111 // Calculate viscous flux components in face normal direction
112 double tau_nn = Txx * nx * nx + Tyy * ny * ny + Tzz * nz * nz +
113 2.0 * (Txy * nx * ny + Txz * nx * nz + Tyz * ny * nz);
114 double tau_nx = Txx * nx + Txy * ny + Txz * nz;
115 double tau_ny = Txy * nx + Tyy * ny + Tyz * nz;
116 double tau_nz = Txz * nx + Tyz * ny + Tzz * nz;
117 double q_n = qx * nx + qy * ny + qz * nz;
118
119 // Get average velocities
120 double u_avg = 0.5 * (primitive_vars[left_cell * 5 + 1]);
121 double v_avg = 0.5 * (primitive_vars[left_cell * 5 + 2]);
122 double w_avg = 0.5 * (primitive_vars[left_cell * 5 + 3]);
123
124 if (right_cell >= 0) {
125 u_avg += 0.5 * primitive_vars[right_cell * 5 + 1];
126 v_avg += 0.5 * primitive_vars[right_cell * 5 + 2];
127 w_avg += 0.5 * primitive_vars[right_cell * 5 + 3];
128 }
129
130 // Assemble viscous fluxes
131 viscous_fluxes[face_idx * 5 + 0] = 0.0; // No mass flux from viscous terms
132 viscous_fluxes[face_idx * 5 + 1] = area * tau_nx;
133 viscous_fluxes[face_idx * 5 + 2] = area * tau_ny;
134 viscous_fluxes[face_idx * 5 + 3] = area * tau_nz;
135 viscous_fluxes[face_idx * 5 + 4] = area * (tau_nx * u_avg + tau_ny * v_avg + tau_nz * w_avg + q_n);
136}

139__global__ void accumulate_cell_viscous_flux_kernel(
140 const double* face_viscous_fluxes, // [num_faces * 5] - Face viscous fluxes
141 const int* cell_faces, // [num_cells * max_faces] - Face indices for each cell
142 const int* num_faces_per_cell, // [num_cells] - Number of faces per cell
143 const int* face_orientations, // [num_cells * max_faces] - Face orientation (+1 or -1)

145 int num_cells,
146 int max_faces
147) {

149 if (cell_idx >= num_cells) return;
150
151 int num_faces = num_faces_per_cell[cell_idx];
152
153 // Initialize cell viscous flux
154 for (int var = 0; var < 5; var++) {
155 cell_viscous_fluxes[cell_idx * 5 + var] = 0.0;
156 }
157
158 // Sum contributions from all faces
159 for (int f = 0; f < num_faces; f++) {
160 int face_id = cell_faces[cell_idx * max_faces + f];
161 int orientation = face_orientations[cell_idx * max_faces + f];
162
163 if (face_id >= 0) {
164 for (int var = 0; var < 5; var++) {
165 cell_viscous_fluxes[cell_idx * 5 + var] +=
166 orientation * face_viscous_fluxes[face_id * 5 + var];
167 }
168 }

173__global__ void artificial_viscosity_kernel(
174 const double* U_cells, // [num_cells * 5] - Conservative variables
175 const double* pressure, // [num_cells] - Pressure field
176 const int* cell_neighbors, // [num_cells * max_neighbors] - Neighbor indices

179 double C2, // Artificial viscosity coefficient

182) {
183 int idx = blockIdx.x * blockDim.x + threadIdx.x;
184 if (idx >= num_cells) return;
185
186 int num_neigh = num_neighbors[idx];
187 double P_center = pressure[idx];
188
189 for (int var = 0; var < 5; var++) {
190 artificial_viscosity[idx * 5 + var] = 0.0;
191 }
192
193 // Calculate pressure-based artificial viscosity
194 double max_pressure_jump = 0.0;
195 for (int n = 0; n < num_neigh; n++) {
196 int neigh_id = cell_neighbors[idx * max_neighbors + n];
197 if (neigh_id >= 0 && neigh_id < num_cells) {
198 double P_neigh = pressure[neigh_id];
199 double pressure_jump = fabs(P_neigh - P_center) / (P_neigh + P_center + 1e-12);
200 max_pressure_jump = fmax(max_pressure_jump, pressure_jump);
201 }
202 }
203
204 // Apply artificial viscosity if pressure jump is significant
205 if (max_pressure_jump > 0.1) {
206 double eps = C2 * max_pressure_jump;
207
208 for (int n = 0; n < num_neigh; n++) {
209 int neigh_id = cell_neighbors[idx * max_neighbors + n];
210 if (neigh_id >= 0 && neigh_id < num_cells) {
211 for (int var = 0; var < 5; var++) {
212 artificial_viscosity[idx * 5 + var] +=
213 eps * (U_cells[neigh_id * 5 + var] - U_cells[idx * 5 + var]);
214 }
215 }
216 }
217 }
