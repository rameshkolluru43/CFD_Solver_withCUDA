161void Runge_Kutta_Method_With_Turbulence()
162{

164 vector<vector<double>> U_initial(Total_Cells, vector<double>(5));
165 vector<TurbulenceVariables> turb_initial(Total_Cells);

170 {

172 turb_initial[i] = turbulence_vars[i];
173 }
174 }
175
176 // RK4 stages
177 for (int stage = 1; stage <= 4; stage++)
178 {
179 // Calculate time step
180 Calculate_Time_Step();
181
182 // Calculate flow residuals

184
185 // Calculate turbulence residuals
186 Calculate_Turbulence_Residuals();

192 Update_Flow_Variables_RK4(U_initial, alpha, beta);

195 // Apply boundary conditions after each stage
196 Apply_Boundary_Conditions();
197 Apply_All_Turbulence_Boundary_Conditions();

283void Calculate_Turbulence_Cell_Residual(int cell_index)
284{
285 if (Cells[cell_index].cellType == GHOST_CELL)
286 return;
287
288 double rho = Cells[cell_index].Conservative_Variables[0];
289 double k = turbulence_vars[cell_index].k;
290 double Pk = turbulence_vars[cell_index].Pk;
291
292 switch (current_turbulence_model)
293 {
294 case TurbulenceModel::K_EPSILON:
295 {
296 double epsilon = turbulence_vars[cell_index].epsilon;
297
298 // K-equation residual
299 double dk_diff = Calculate_KEpsilon_K_Diffusion(cell_index);
300 double Rk = dk_diff + Pk - rho * epsilon;
301
302 // Epsilon-equation residual
303 double depsilon_diff = Calculate_KEpsilon_Epsilon_Diffusion(cell_index);
304 double Repsilon = depsilon_diff + KEpsilon::C_1 * (epsilon / k) * Pk -
305 KEpsilon::C_2 * rho * (epsilon * epsilon / k);
306
307 // Store residuals (these would be used in implicit solver)
308 Cells[cell_index].k_residual = Rk;
309 Cells[cell_index].epsilon_residual = Repsilon;
310 break;
311 }
312
313 case TurbulenceModel::K_OMEGA_WILCOX:
314 case TurbulenceModel::K_OMEGA_SST:
315 {
316 double omega = turbulence_vars[cell_index].omega;
317
318 // K-equation residual
319 double dk_diff = Calculate_KOmega_K_Diffusion(cell_index);
320 double Rk = dk_diff + Pk - KOmega::beta_star * rho * k * omega;
321
322 // Omega-equation residual
323 double domega_diff = Calculate_KOmega_Omega_Diffusion(cell_index);
324 double Romega = domega_diff + KOmega::alpha * (omega / k) * Pk -
325 KOmega::beta * rho * omega * omega;
326
327 // Store residuals
328 Cells[cell_index].k_residual = Rk;
329 Cells[cell_index].omega_residual = Romega;
330 break;
331 }
332
333 default:
334 break;
335 }
336}

348void Update_Turbulence_Variables_RK4(const vector<TurbulenceVariables> &turb_initial,
349 double alpha, double beta)
350{
351 if (current_turbulence_model == TurbulenceModel::LAMINAR)
352 return;
353
354 for (int i = 0; i < Total_Cells; i++)
355 {
356 if (Cells[i].cellType != GHOST_CELL)
357 {
358 double rho = Cells[i].Conservative_Variables[0];
359 double dt = Min_dt;
360
361 // Update k
362 double k_old = turb_initial[i].k;
363 double k_residual = Cells[i].k_residual;
364 double k_new = k_old + alpha * dt * k_residual / rho;
365 turbulence_vars[i].k = max(k_new, 1e-10);
366
367 // Update epsilon or omega
368 if (current_turbulence_model == TurbulenceModel::K_EPSILON)
369 {
370 double eps_old = turb_initial[i].epsilon;
371 double eps_residual = Cells[i].epsilon_residual;
372 double eps_new = eps_old + alpha * dt * eps_residual / rho;
373 turbulence_vars[i].epsilon = max(eps_new, 1e-10);
374
375 Calculate_KEpsilon_Turbulent_Viscosity(i);
376 }
377 else
378 {
379 double omega_old = turb_initial[i].omega;
380 double omega_residual = Cells[i].omega_residual;
381 double omega_new = omega_old + alpha * dt * omega_residual / rho;
382 turbulence_vars[i].omega = max(omega_new, 1e-6);
383
384 Calculate_KOmega_Turbulent_Viscosity(i);
385 }
386 }
387 }

399double Calculate_Turbulence_Error(const string &variable)
400{
401 if (current_turbulence_model == TurbulenceModel::LAMINAR)
402 return 0.0;
403
404 double error_sum = 0.0;
405 int count = 0;
406
407 for (int i = 0; i < Total_Cells; i++)
408 {
409 if (Cells[i].cellType != GHOST_CELL)
410 {
411 double error = 0.0;
412
413 if (variable == "k")
414 {
415 error = abs(Cells[i].k_residual);
416 }
417 else if (variable == "eps_omega")
418 {
419 if (current_turbulence_model == TurbulenceModel::K_EPSILON)
420 {
421 error = abs(Cells[i].epsilon_residual);
422 }
423 else
424 {
425 error = abs(Cells[i].omega_residual);
426 }
427 }
428
429 error_sum += error * error;
430 count++;
431 }
432 }
433
434 return (count > 0) ? sqrt(error_sum / count) : 0.0;
435}

479void Apply_Turbulence_Limiters(int cell_index)
480{
481 if (Cells[cell_index].cellType == GHOST_CELL)
482 return;
483
484 // Apply positivity constraints
485 turbulence_vars[cell_index].k = max(turbulence_vars[cell_index].k, 1e-10);
486
487 if (current_turbulence_model == TurbulenceModel::K_EPSILON)
488 {
489 turbulence_vars[cell_index].epsilon = max(turbulence_vars[cell_index].epsilon, 1e-10);
490 }
491 else
492 {
493 turbulence_vars[cell_index].omega = max(turbulence_vars[cell_index].omega, 1e-6);
494 }
495
496 // Apply realizability constraints
497 Apply_Realizability_Constraints(cell_index);
498
499 // Limit turbulent viscosity ratio
500 double mu_laminar = Calculate_Laminar_Viscosity(cell_index);
501 double max_turbulent_viscosity = 1000.0 * mu_laminar;
502
503 if (turbulence_vars[cell_index].mut > max_turbulent_viscosity)
504 {
505 turbulence_vars[cell_index].mut = max_turbulent_viscosity;
506 turbulence_vars[cell_index].nut = turbulence_vars[cell_index].mut /
507 Cells[cell_index].Conservative_Variables[0];
508 }
