3//Gets the first gradient and calculates Del2_Q_1,Del2_Q_2,Del2_Q_3
4void Face::Cal_Del2Q(Cell * Current_Cell,Cell* Neighbour_Cell)
5{
6 Del_Q_1[0]=0.0;Del_Q_1[1]=0.0;Del_Q_1[2]=0.0;Del_Q_1[3]=0.0;Del_Q_1[4]=0.0;
7 Del_Q_2[0]=0.0;Del_Q_2[1]=0.0;Del_Q_2[2]=0.0;Del_Q_2[3]=0.0;Del_Q_2[4]=0.0;
8 Del_Q_3[0]=0.0;Del_Q_3[1]=0.0;Del_Q_3[2]=0.0;Del_Q_3[3]=0.0;Del_Q_3[4]=0.0;
9
10 if(Neighbour_Cell->Is_Ghost_Cell())
11 {
12 rho_ndS = Current_Cell->Get_Q_Grad(0);
13 rhou_ndS = Current_Cell->Get_Q_Grad(1);
14 rhov_ndS = Current_Cell->Get_Q_Grad(2);
15 rhow_ndS = Current_Cell->Get_Q_Grad(3);
16 rhoET_ndS = Current_Cell->Get_Q_Grad(4);
17 }
18 else
19 {
20 rho_ndS = (Current_Cell->Get_Q_Grad(0) + Neighbour_Cell->Get_Q_Grad(0))*0.5;
21 rhou_ndS = (Current_Cell->Get_Q_Grad(1) + Neighbour_Cell->Get_Q_Grad(1))*0.5;
22 rhov_ndS = (Current_Cell->Get_Q_Grad(2) + Neighbour_Cell->Get_Q_Grad(2))*0.5;
23 rhow_ndS = (Current_Cell->Get_Q_Grad(3) + Neighbour_Cell->Get_Q_Grad(3))*0.5;
24 rhoET_ndS = (Current_Cell->Get_Q_Grad(4) + Neighbour_Cell->Get_Q_Grad(4))*0.5;
25 }
26
27 Del_Q_1[0] = rho_ndS(1)*Area_Comp1;
28 Del_Q_2[0] = rho_ndS(2)*Area_Comp2;
29 Del_Q_3[0] = rho_ndS(3)*Area_Comp3;
30
31 Del_Q_1[1] = rhou_ndS(1)*Area_Comp1;
32 Del_Q_2[1] = rhou_ndS(2)*Area_Comp2;
33 Del_Q_3[1] = rhou_ndS(3)*Area_Comp3;
34
35 Del_Q_1[2] = rhov_ndS(1)*Area_Comp1;
36 Del_Q_2[2] = rhov_ndS(2)*Area_Comp2;
37 Del_Q_3[2] = rhov_ndS(3)*Area_Comp3;
38
39 Del_Q_1[3] = rhow_ndS(1)*Area_Comp1;
40 Del_Q_2[3] = rhow_ndS(2)*Area_Comp2;
41 Del_Q_3[3] = rhow_ndS(3)*Area_Comp3;
42
43 Del_Q_1[4] = rhoET_ndS(1)*Area_Comp1;
44 Del_Q_2[4] = rhoET_ndS(2)*Area_Comp2;
45 Del_Q_3[4] = rhoET_ndS(3)*Area_Comp3;
46}

50const vector<double>& Face::Calculate_Grad_n_Q(const Cell * const CC, const Cell * const NC, const int &n, const int & i)
51{
52 Del_Q_1[0]=0.0;Del_Q_1[1]=0.0;Del_Q_1[2]=0.0;Del_Q_1[3]=0.0;Del_Q_1[4]=0.0;
53 Del_Q_2[0]=0.0;Del_Q_2[1]=0.0;Del_Q_2[2]=0.0;Del_Q_2[3]=0.0;Del_Q_2[4]=0.0;
54 Del_Q_3[0]=0.0;Del_Q_3[1]=0.0;Del_Q_3[2]=0.0;Del_Q_3[3]=0.0;Del_Q_3[4]=0.0;
55
56 vector<double> gradq1(5,0.0),grad_avg(5,0.0),gradq2(5,0.0);
57 // n refers for the nth gradient and i refers to the direction
58 gradq1 = CC->Get_DelQ_fromCell(n,i);
59 if(NC->Is_Ghost_Cell())
60 gradq2 = gradq1;
61 else
62 gradq2 = NC->Get_DelQ_fromCell(n,i);
63
64 for(int k=0;k<5;k++)
65 grad_avg[k] = 0.5*(gradq1[k]+gradq2[k]);
66
67 switch(i)
68 {
69 case 1:
70 for(int p=0;p<5;p++)
71 Del_Q_1[p] = grad_avg[p]*Area_Comp1;
72 return Del_Q_1;
73 case 2:
74 for(int p=0;p<5;p++)
75 Del_Q_2[p] = grad_avg[p]*Area_Comp2;
76 return Del_Q_2;
77 case 3:
78 for(int p=0;p<5;p++)
79 Del_Q_3[p] = grad_avg[p]*Area_Comp3;
80 return Del_Q_3;
81 default:
82 cout<<"Check of i , should be 1,2,3 to get the corresponding gradients\n";
83 break;
84 }
