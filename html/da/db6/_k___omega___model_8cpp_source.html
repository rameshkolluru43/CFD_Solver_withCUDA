74void Calculate_KOmega_Turbulent_Viscosity(int cell_index)
75{
76 if (Cells[cell_index].cellType == GHOST_CELL)
77 return;
78
79 double k = turbulence_vars[cell_index].k;
80 double omega = turbulence_vars[cell_index].omega;
81 double rho = Cells[cell_index].Conservative_Variables[0];
82
83 // Ensure positive values
84 k = max(k, 1e-10);
85 omega = max(omega, 1e-6);
86
87 if (current_turbulence_model == TurbulenceModel::K_OMEGA_SST)
88 {
89 // SST model: μt = ρa₁k/max(a₁ω, ΩF₂)
90 double vorticity_magnitude = Calculate_Vorticity_Magnitude(cell_index);
91 double F2 = Calculate_SST_F2_Function(cell_index);
92
93 double denominator = max(SST::a1 * omega, vorticity_magnitude * F2);
94 turbulence_vars[cell_index].mut = rho * SST::a1 * k / denominator;
95 }
96 else
97 {
98 // Standard K-omega: μt = ρk/ω
99 turbulence_vars[cell_index].mut = rho * k / omega;
100 }
101
102 turbulence_vars[cell_index].nut = turbulence_vars[cell_index].mut / rho;
103
104 // Apply realizability constraint
105 double S_magnitude = Calculate_Strain_Rate_Magnitude(cell_index);
106 if (S_magnitude > 0)
107 {
108 double realizability_limit = 2.0 * k / S_magnitude;
109 turbulence_vars[cell_index].nut = min(turbulence_vars[cell_index].nut, realizability_limit);
110 turbulence_vars[cell_index].mut = turbulence_vars[cell_index].nut * rho;
111 }
112
113 // Limit maximum turbulent viscosity
114 double mu_laminar = Calculate_Laminar_Viscosity(cell_index);
115 turbulence_vars[cell_index].mut = min(turbulence_vars[cell_index].mut, 1000.0 * mu_laminar);
116}

126void Calculate_KOmega_Production_Terms(int cell_index)
127{
128 if (Cells[cell_index].cellType == GHOST_CELL)
129 return;
130
131 // Calculate strain rate tensor
132 double S[3][3];
133 Calculate_Strain_Rate_Tensor(cell_index, S);
134
135 // Calculate strain rate magnitude
136 double S_magnitude = 0.0;
137 for (int i = 0; i < 3; i++)
138 {
139 for (int j = 0; j < 3; j++)
140 {
141 S_magnitude += S[i][j] * S[i][j];
142 }
143 }
144 S_magnitude = sqrt(2.0 * S_magnitude);
145
146 // Calculate production of turbulent kinetic energy
147 double mut = turbulence_vars[cell_index].mut;
148 turbulence_vars[cell_index].Pk = mut * S_magnitude * S_magnitude;
149
150 // Limit production to prevent excessive values
151 double k = turbulence_vars[cell_index].k;
152 double omega = turbulence_vars[cell_index].omega;
153 double rho = Cells[cell_index].Conservative_Variables[0];
154
155 if (current_turbulence_model == TurbulenceModel::K_OMEGA_SST)
156 {
157 // SST production limiter
158 double max_production = 10.0 * KOmega::beta_star * rho * k * omega;
159 turbulence_vars[cell_index].Pk = min(turbulence_vars[cell_index].Pk, max_production);
160 }
161 else
162 {
163 // Standard K-omega production limiter
164 double max_production = 10.0 * KOmega::beta * rho * k * omega;
165 turbulence_vars[cell_index].Pk = min(turbulence_vars[cell_index].Pk, max_production);
166 }
167}

178void Solve_KOmega_Transport_Equations(int cell_index, double dt)
179{
180 if (Cells[cell_index].cellType == GHOST_CELL)
181 return;
182
183 double rho = Cells[cell_index].Conservative_Variables[0];
184 double k = turbulence_vars[cell_index].k;
185 double omega = turbulence_vars[cell_index].omega;
186 double Pk = turbulence_vars[cell_index].Pk;
187
188 // Calculate diffusion terms
189 double dk_diff = Calculate_KOmega_K_Diffusion(cell_index);
190 double domega_diff = Calculate_KOmega_Omega_Diffusion(cell_index);
191
192 if (current_turbulence_model == TurbulenceModel::K_OMEGA_SST)
193 {
194 Solve_SST_Transport_Equations(cell_index, dt);
195 return;
196 }
197
198 // Standard K-omega model
199 // K-equation source terms
200 double Sk = Pk - KOmega::beta_star * rho * k * omega;
201
202 // Omega-equation source terms
203 double Somega = KOmega::alpha * (omega / k) * Pk - KOmega::beta * rho * omega * omega;
204
205 // Time integration (explicit Euler)
206 double k_new = k + dt * (dk_diff + Sk) / rho;
207 double omega_new = omega + dt * (domega_diff + Somega) / rho;
208
209 // Apply positivity constraints
210 turbulence_vars[cell_index].k = max(k_new, 1e-10);
211 turbulence_vars[cell_index].omega = max(omega_new, 1e-6);
212
213 // Update turbulent viscosity
214 Calculate_KOmega_Turbulent_Viscosity(cell_index);
215}

226void Solve_SST_Transport_Equations(int cell_index, double dt)
227{
228 if (Cells[cell_index].cellType == GHOST_CELL)
229 return;
230
231 double rho = Cells[cell_index].Conservative_Variables[0];
232 double k = turbulence_vars[cell_index].k;
233 double omega = turbulence_vars[cell_index].omega;
234 double Pk = turbulence_vars[cell_index].Pk;
235
236 // Calculate blending functions
237 double F1, F2;
238 Calculate_SST_Blending_Functions(cell_index, F1, F2);
239
240 // Blended constants
241 double sigma_k = F1 * SST::sigma_k1 + (1.0 - F1) * SST::sigma_k2;
242 double sigma_omega = F1 * SST::sigma_w1 + (1.0 - F1) * SST::sigma_w2;
243 double alpha = F1 * SST::alpha_1 + (1.0 - F1) * SST::alpha_2;
244 double beta = F1 * SST::beta_1 + (1.0 - F1) * SST::beta_2;
245
246 // Calculate diffusion terms with blended constants
247 double dk_diff = Calculate_SST_K_Diffusion(cell_index, sigma_k);
248 double domega_diff = Calculate_SST_Omega_Diffusion(cell_index, sigma_omega);
249
250 // Cross-diffusion term for omega equation
251 double CDkomega = Calculate_SST_Cross_Diffusion(cell_index);
252
253 // Source terms
254 double Sk = Pk - SST::beta_star * rho * k * omega;
255 double Somega = alpha * (omega / k) * Pk - beta * rho * omega * omega +
256 2.0 * (1.0 - F1) * SST::sigma_w2 * rho * CDkomega;
257
258 // Time integration
259 double k_new = k + dt * (dk_diff + Sk) / rho;
260 double omega_new = omega + dt * (domega_diff + Somega) / rho;
261
262 // Apply constraints
263 turbulence_vars[cell_index].k = max(k_new, 1e-10);
264 turbulence_vars[cell_index].omega = max(omega_new, 1e-6);
265
266 // Update turbulent viscosity
267 Calculate_KOmega_Turbulent_Viscosity(cell_index);
268}

276void Calculate_SST_Blending_Functions(int cell_index, double &F1, double &F2)
277{
278 double k = turbulence_vars[cell_index].k;
279 double omega = turbulence_vars[cell_index].omega;
280 double rho = Cells[cell_index].Conservative_Variables[0];
281 double mu = Calculate_Laminar_Viscosity(cell_index);
282
283 double y_wall = turbulence_vars[cell_index].y_wall;
284
285 // Calculate CDkomega
286 double CDkomega = Calculate_SST_Cross_Diffusion(cell_index);
287 CDkomega = max(CDkomega, 1e-20);
288
289 // Calculate arguments for F1
290 double arg1_1 = sqrt(k) / (SST::beta_star * omega * y_wall);
291 double arg1_2 = 500.0 * mu / (rho * omega * y_wall * y_wall);
292 double arg1_3 = 4.0 * SST::sigma_w2 * rho * k / (CDkomega * y_wall * y_wall);
293
294 double arg1 = min(max(arg1_1, arg1_2), arg1_3);
295 F1 = tanh(pow(arg1, 4));
296
297 // Calculate argument for F2
298 double arg2_1 = 2.0 * sqrt(k) / (SST::beta_star * omega * y_wall);
299 double arg2_2 = 500.0 * mu / (rho * omega * y_wall * y_wall);
300
301 double arg2 = max(arg2_1, arg2_2);
302 F2 = tanh(pow(arg2, 2));
303}

310double Calculate_SST_Cross_Diffusion(int cell_index)
311{
312 double CDkomega = 0.0;
313
314 // Calculate gradients of k and omega
315 double grad_k[3], grad_omega[3];
316 Calculate_Scalar_Gradient(cell_index, "k", grad_k);
317 Calculate_Scalar_Gradient(cell_index, "omega", grad_omega);
318
319 // CDkomega = (∇k · ∇ω) / ω
320 double omega = turbulence_vars[cell_index].omega;
321 for (int i = 0; i < 3; i++)
322 {
323 CDkomega += grad_k[i] * grad_omega[i];
324 }
325 CDkomega = CDkomega / omega;
326
327 return max(CDkomega, 0.0);
328}

339double Calculate_KOmega_K_Diffusion(int cell_index)
340{
341 if (Cells[cell_index].cellType == GHOST_CELL)
342 return 0.0;
343
344 double diffusion = 0.0;
345 double mu_laminar = Calculate_Laminar_Viscosity(cell_index);
346 double mut = turbulence_vars[cell_index].mut;
347 double mu_eff = mu_laminar + KOmega::sigma_star * mut;
348
349 // Face-based diffusion calculation
350 for (int face = 0; face < Cells[cell_index].No_of_Faces; face++)
351 {
352 int neighbor = Cells[cell_index].Neighbours[face];
353 if (neighbor >= 0 && neighbor < Total_Cells)
354 {
355 double k_neighbor = turbulence_vars[neighbor].k;
356 double k_cell = turbulence_vars[cell_index].k;
357
358 double face_area = Cells[cell_index].Face_Areas[face];
359 double distance = Calculate_Cell_Distance(cell_index, neighbor);
360
361 diffusion += mu_eff * (k_neighbor - k_cell) * face_area / distance;
362 }
363 }
364
365 return diffusion;
366}

410void Apply_KOmega_Wall_Functions(int cell_index)
411{
412 if (!Cells[cell_index].hasBoundaryface)
413 return;
414
415 // Calculate wall parameters
416 Calculate_Wall_Distance(cell_index);
417 Calculate_Wall_Shear_Stress(cell_index);
418 Calculate_Y_Plus(cell_index);
419
420 double y_plus = turbulence_vars[cell_index].y_plus;
421 double u_tau = turbulence_vars[cell_index].u_tau;
422 double y_wall = turbulence_vars[cell_index].y_wall;
423 double rho = Cells[cell_index].Conservative_Variables[0];
424 double mu = Calculate_Laminar_Viscosity(cell_index);
425
426 if (y_plus > 30.0)
427 { // Log-law region
428 // K at the wall
429 double k_wall = u_tau * u_tau / sqrt(KOmega::beta_star);
430
431 // Omega at the wall using log-law
432 double omega_wall = u_tau / (sqrt(KOmega::beta_star) * KOmega::kappa * y_wall);
433
434 turbulence_vars[cell_index].k = k_wall;
435 turbulence_vars[cell_index].omega = omega_wall;
436 }
437 else if (y_plus < 1.0)
438 { // Viscous sublayer
439 // Near-wall treatment
440 turbulence_vars[cell_index].k = 0.0;
441
442 // Omega in viscous sublayer
443 double omega_viscous = 60.0 * mu / (rho * KOmega::beta * y_wall * y_wall);
444 turbulence_vars[cell_index].omega = omega_viscous;
445 }
446
447 // Update turbulent viscosity
448 Calculate_KOmega_Turbulent_Viscosity(cell_index);
449}

536void Calculate_Scalar_Gradient(int cell_index, const string &variable, double grad[3])
537{
538 // Initialize gradient
539 grad[0] = grad[1] = grad[2] = 0.0;
540
541 double value_center;
542 if (variable == "k")
543 {
544 value_center = turbulence_vars[cell_index].k;
545 }
546 else if (variable == "omega")
547 {
548 value_center = turbulence_vars[cell_index].omega;
549 }
550 else
551 {
552 return;
553 }
554
555 // Green-Gauss gradient calculation
556 for (int face = 0; face < Cells[cell_index].No_of_Faces; face++)
557 {
558 int neighbor = Cells[cell_index].Neighbours[face];
559 double value_neighbor;
560
561 if (neighbor >= 0 && neighbor < Total_Cells)
562 {
563 if (variable == "k")
564 {
565 value_neighbor = turbulence_vars[neighbor].k;
566 }
567 else
568 {
569 value_neighbor = turbulence_vars[neighbor].omega;
570 }
571 }
572 else
573 {
574 value_neighbor = value_center; // Zero gradient at boundary
575 }
576
577 double face_value = 0.5 * (value_center + value_neighbor);
578 double face_area = Cells[cell_index].Face_Areas[face];
579
580 // Face normal components (assuming they are stored)
581 double nx = Cells[cell_index].Face_Normals[face][0];
582 double ny = Cells[cell_index].Face_Normals[face][1];
583 double nz = Cells[cell_index].Face_Normals[face][2];
584
585 grad[0] += face_value * nx * face_area;
586 grad[1] += face_value * ny * face_area;
587 grad[2] += face_value * nz * face_area;
588 }
589
590 // Divide by cell volume
591 double cell_volume = Cells[cell_index].Volume;
592 grad[0] /= cell_volume;
593 grad[1] /= cell_volume;
594 grad[2] /= cell_volume;
595}
