109 {
110 inv_eta = 1.0 / (eta * j);
111 inv_zi = 1.0 / (zi * i);
112
113 xzi = 0.5 * inv_zi * (x[i + 1][j] - xtemp[i - 1][j]);
114 xeta = 0.5 * inv_eta * (x[i][j + 1] - xtemp[i][j - 1]);
115 yzi = 0.5 * inv_zi * (y[i + 1][j] - ytemp[i - 1][j]);
116 yeta = 0.5 * inv_eta * (y[i][j + 1] - ytemp[i][j - 1]);
117
118 alpha = (xeta * xeta + yeta * yeta); //(g22)i,j alpha
119 gamma1 = (xzi * xzi + yzi * yzi); //(g11)i,j gamma
120 beta = (xzi * xeta + yzi * yeta); //(g12)i,j beta
121
122 B = alpha * inv_zi * inv_zi;
123 D = gamma1 * inv_eta * inv_eta;
124 C = -0.5 * beta * inv_eta * inv_zi;
125 A = B + D;
