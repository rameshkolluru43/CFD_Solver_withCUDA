71void Calculate_KEpsilon_Turbulent_Viscosity(int cell_index)
72{
73 if (Cells[cell_index].cellType == GHOST_CELL)
74 return;
75
76 double k = turbulence_vars[cell_index].k;
77 double epsilon = turbulence_vars[cell_index].epsilon;
78 double rho = Cells[cell_index].Conservative_Variables[0];
79
80 // Ensure positive values
81 k = max(k, 1e-10);
82 epsilon = max(epsilon, 1e-10);
83
84 // Calculate turbulent viscosity: μt = ρCμk²/ε
85 turbulence_vars[cell_index].mut = rho * KEpsilon::C_mu * k * k / epsilon;
86 turbulence_vars[cell_index].nut = turbulence_vars[cell_index].mut / rho;
87
88 // Apply realizability constraint
89 double S_magnitude = Calculate_Strain_Rate_Magnitude(cell_index);
90 if (S_magnitude > 0)
91 {
92 double realizability_limit = 2.0 * k / S_magnitude;
93 turbulence_vars[cell_index].nut = min(turbulence_vars[cell_index].nut, realizability_limit);
94 turbulence_vars[cell_index].mut = turbulence_vars[cell_index].nut * rho;
95 }
96
97 // Limit maximum turbulent viscosity
98 double mu_laminar = Calculate_Laminar_Viscosity(cell_index);
99 turbulence_vars[cell_index].mut = min(turbulence_vars[cell_index].mut, 1000.0 * mu_laminar);
100}

110void Calculate_KEpsilon_Production_Terms(int cell_index)
111{
112 if (Cells[cell_index].cellType == GHOST_CELL)
113 return;
114
115 // Calculate strain rate tensor components
116 double S[3][3];
117 Calculate_Strain_Rate_Tensor(cell_index, S);
118
119 // Calculate strain rate magnitude: S = sqrt(2SijSij)
120 double S_magnitude = 0.0;
121 for (int i = 0; i < 3; i++)
122 {
123 for (int j = 0; j < 3; j++)
124 {
125 S_magnitude += S[i][j] * S[i][j];
126 }
127 }
128 S_magnitude = sqrt(2.0 * S_magnitude);
129
130 // Calculate production of turbulent kinetic energy: Pk = μt * S²
131 double mut = turbulence_vars[cell_index].mut;
132 turbulence_vars[cell_index].Pk = mut * S_magnitude * S_magnitude;
133
134 // Limit production to prevent excessive values
135 double k = turbulence_vars[cell_index].k;
136 double epsilon = turbulence_vars[cell_index].epsilon;
137 double rho = Cells[cell_index].Conservative_Variables[0];
138
139 // Apply Kato-Launder correction if needed
140 double max_production = 10.0 * rho * epsilon;
141 turbulence_vars[cell_index].Pk = min(turbulence_vars[cell_index].Pk, max_production);
142}

153void Solve_KEpsilon_Transport_Equations(int cell_index, double dt)
154{
155 if (Cells[cell_index].cellType == GHOST_CELL)
156 return;
157
158 double rho = Cells[cell_index].Conservative_Variables[0];
159 double k = turbulence_vars[cell_index].k;
160 double epsilon = turbulence_vars[cell_index].epsilon;
161 double Pk = turbulence_vars[cell_index].Pk;
162
163 // Calculate diffusion terms (simplified for structured grids)
164 double dk_diff = Calculate_KEpsilon_K_Diffusion(cell_index);
165 double depsilon_diff = Calculate_KEpsilon_Epsilon_Diffusion(cell_index);
166
167 // K-equation source terms
168 double Sk = Pk - rho * epsilon;
169
170 // Epsilon-equation source terms
171 double Sepsilon = KEpsilon::C_1 * (epsilon / k) * Pk - KEpsilon::C_2 * rho * (epsilon * epsilon / k);
172
173 // Time integration (explicit Euler)
174 double k_new = k + dt * (dk_diff + Sk) / rho;
175 double epsilon_new = epsilon + dt * (depsilon_diff + Sepsilon) / rho;
176
177 // Apply positivity constraints
178 turbulence_vars[cell_index].k = max(k_new, 1e-10);
179 turbulence_vars[cell_index].epsilon = max(epsilon_new, 1e-10);
180
181 // Update turbulent viscosity
182 Calculate_KEpsilon_Turbulent_Viscosity(cell_index);
183}

194double Calculate_KEpsilon_K_Diffusion(int cell_index)
195{
196 if (Cells[cell_index].cellType == GHOST_CELL)
197 return 0.0;
198
199 double diffusion = 0.0;
200 double mu_laminar = Calculate_Laminar_Viscosity(cell_index);
201 double mut = turbulence_vars[cell_index].mut;
202 double mu_eff = mu_laminar + mut / KEpsilon::sigma_k;
203
204 // Calculate diffusion using face-based approach
205 for (int face = 0; face < Cells[cell_index].No_of_Faces; face++)
206 {
207 int neighbor = Cells[cell_index].Neighbours[face];
208 if (neighbor >= 0 && neighbor < Total_Cells)
209 {
210 double k_neighbor = turbulence_vars[neighbor].k;
211 double k_cell = turbulence_vars[cell_index].k;
212
213 double face_area = Cells[cell_index].Face_Areas[face];
214 double distance = Calculate_Cell_Distance(cell_index, neighbor);
215
216 diffusion += mu_eff * (k_neighbor - k_cell) * face_area / distance;
217 }
218 }
219
220 return diffusion;
221}

377double Calculate_Laminar_Viscosity(int cell_index)
378{
379 double rho = Cells[cell_index].Conservative_Variables[0];
380 double E = Cells[cell_index].Conservative_Variables[4];
381 double u = Cells[cell_index].Conservative_Variables[1] / rho;
382 double v = Cells[cell_index].Conservative_Variables[2] / rho;
383 double w = Cells[cell_index].Conservative_Variables[3] / rho;
384
385 double kinetic_energy = 0.5 * (u * u + v * v + w * w);
386 double internal_energy = E / rho - kinetic_energy;
387 double T = internal_energy / cv;
388
389 // Sutherland's law
390 return C_1 * pow(T, 1.5) / (T + T_S_Mu);
391}
