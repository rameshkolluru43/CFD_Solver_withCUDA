63// Function to convert all triangles to quadrilateral mesh
64vector<Quad> convertAllToQuads(const vector<Triangle>& triangles) {
65 unordered_map<pair<int, int>, vector<int>, hash_pair> edgeMap;
66 vector<Quad> quads;
67 vector<bool> used(triangles.size(), false);
68
69 // Populate edge map with triangle indices
70 for (size_t i = 0; i < triangles.size(); ++i) {
71 for (int j = 0; j < 3; ++j) {
72 int n1 = triangles[i][j];
73 int n2 = triangles[i][(j + 1) % 3];
74 if (n1 > n2) swap(n1, n2);
75 edgeMap[make_pair(n1, n2)].push_back(i);
76 }
77 }
78
79 // Find adjacent triangles that share an edge and convert all to quads
80 for (auto& entry : edgeMap) {
81 vector<int>& triIndices = entry.second;
82 if (triIndices.size() == 2) { // Shared by two triangles
83 int t1 = triIndices[0];
84 int t2 = triIndices[1];
85 if (!used[t1] && !used[t2]) {
86 vector<int> q = {triangles[t1][0], triangles[t1][1], triangles[t2][1], triangles[t2][2]};
87 quads.push_back(q);
88 used[t1] = used[t2] = true;
89 }
90 }
91 }
92
93 return quads;
94}
