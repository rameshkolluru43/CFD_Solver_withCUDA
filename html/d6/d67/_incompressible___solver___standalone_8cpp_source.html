270double IncompressibleFlowSolver::calculate_residual()
271{
272 double residual = 0.0;
273
274 for (int i = 0; i < num_cells; ++i)
275 {
276 double du = velocity.u[i] - velocity.u_old[i];
277 double dv = velocity.v[i] - velocity.v_old[i];
278 double dp = pressure.p[i] - pressure.p_old[i];
279
280 residual += du * du + dv * dv + dp * dp;
281 if (!is_2d)
282 {
283 double dw = velocity.w[i] - velocity.w_old[i];
284 residual += dw * dw;
285 }
286 }
287
288 return std::sqrt(residual / num_cells);
289}
