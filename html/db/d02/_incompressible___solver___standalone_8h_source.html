134// Fluid properties
135struct FluidProperties
136{
137 double density; // kg/m³
138 double viscosity; // Pa⋅s (dynamic viscosity)
139 double kinematic_viscosity; // m²/s (= viscosity/density)
140
141 FluidProperties(double rho = 1.0, double mu = 1e-3)
142 : density(rho), viscosity(mu), kinematic_viscosity(mu / rho) {}
143
144 void update()
145 {
146 kinematic_viscosity = viscosity / density;
147 }
148
149 double reynolds_number(double characteristic_velocity, double characteristic_length) const
150 {
151 return (density * characteristic_velocity * characteristic_length) / viscosity;
152 }
153};
