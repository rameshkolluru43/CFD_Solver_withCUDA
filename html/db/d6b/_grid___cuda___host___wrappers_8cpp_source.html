405// Estimate memory requirements for grid operations

407 int max_faces_per_cell, int max_neighbors)
408{
409 size_t total_memory = 0;
410
411 // Point coordinates
412 total_memory += num_points * 3 * sizeof(double);
413
414 // Cell data
415 total_memory += num_cells * sizeof(double); // Areas
416 total_memory += num_cells * 3 * sizeof(double); // Centers
417 total_memory += num_cells * max_faces_per_cell * sizeof(int); // Face connectivity
418 total_memory += num_cells * max_neighbors * sizeof(int); // Neighbors
419
420 // Face data
421 total_memory += num_faces * sizeof(double); // Areas
422 total_memory += num_faces * 3 * sizeof(double); // Centers
423 total_memory += num_faces * 3 * sizeof(double); // Normals
424 total_memory += num_faces * 2 * sizeof(int); // Cell connectivity
425
426 // Quality metrics
427 total_memory += num_cells * 3 * sizeof(double); // Aspect ratio, skewness, orthogonality
428
429 return total_memory;
430}

432// Validate grid data consistency
433bool validate_grid_data(int num_points, int num_cells, int num_faces,
434 const int *cell_connectivity, const int *cell_offsets)
435{
436 if (num_points <= 0 || num_cells <= 0 || num_faces <= 0)
437 {
438 std::cerr << "Invalid grid dimensions" << std::endl;
439 return false;
440 }
441
442 if (cell_connectivity == nullptr || cell_offsets == nullptr)
443 {
444 std::cerr << "Null connectivity data" << std::endl;
445 return false;
446 }
447
448 // Check connectivity indices are within bounds
449 for (int i = 0; i < num_cells; i++)
450 {
451 int start = cell_offsets[i];
452 int end = cell_offsets[i + 1];
453
454 for (int j = start; j < end; j++)
455 {
456 if (cell_connectivity[j] < 0 || cell_connectivity[j] >= num_points)
457 {
458 std::cerr << "Invalid point index in connectivity: " << cell_connectivity[j] << std::endl;
459 return false;
460 }
461 }
462 }
463
464 return true;
465}
